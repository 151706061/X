 13 /**
 14  * The superclass class for all X.base-objects.
 15  *
 16  * @constructor
 17  */
 18 X.base = function() {
 19
 20   //
 21   // class attributes
 22
 23   /**
 24    * The className of this class.
 25    *
 26    * @type {string}
 27    * @protected
 28    */
 29   this._className = 'base';
 30
 31 };

 34 /**
 35  * Print the className of the current X.base-object.
 36  *
 37  * @return {string} The className of the current X.object.
 38  */
 39 X.base.prototype.getClassName = function() {
 40
 41   return this._className;
 42
 43 };

 46 /**
 47  * Print the className and all attributes of the current X.base-object.
 48  * We should use getAttributeName rather than accessing the attribute directly
 49  * After compilation, we can not access to the variables like that....!
 50  * since the variables are renamed (_className -> a )
 51  *
 52  * @return {string} A string representation of the current X.base-object.
 53  */
 54 X.base.prototype.print = function() {
 55
 56   var attributes = Object.keys(this);
 57   var attributesStringList = '';
 58   var className = '<unknown>';
 59
 60   // loop through the attributes of a class
 61   for (var a in attributes) {
 62
 63     var aName = 'this.' + attributes[a];
 64     var aValue = eval(aName);
 65
 66     // catch the className, since we want to display it differently
 67     if (aName == 'this._className') {
 68       className = aValue;
 69     } else {
 70       attributesStringList += aName + ': ' + aValue + '\n';
 71     }
 72
 73   }
 74
 75   // build string output
 76   var output = '== X.' + className + ' ==\n';
 77   output += attributesStringList;
 78
 79   return output;
 80
 81 };
