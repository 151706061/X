 15 /**
 16  * Create a renderer with the given width and height.
 17  *
 18  * @param {number} width The width of the renderer.
 19  * @param {number} height The height of the renderer.
 20  * @constructor
 21  * @extends {X.base}
 22  */
 23 X.renderer = function(width, height) {
 24 
 25   // call the standard constructor of X.base
 26   goog.base(this);
 27 
 28   //
 29   // class attributes
 30 
 31   /**
 32    * @inheritDoc
 33    * @const
 34    */
 35   this._className = 'renderer';
 36 
 37   /**
 38    * The dimension of this renderer.
 39    *
 40    * @type {!number}
 41    * @protected
 42    */
 43   this._dimension = -1;
 44 
 45   /**
 46    * The width of this renderer.
 47    *
 48    * @type {!number}
 49    * @protected
 50    */
 51   this._width = width;
 52 
 53   /**
 54    * The height of this renderer.
 55    *
 56    * @type {!number}
 57    * @protected
 58    */
 59   this._height = height;
 60 
 61   /**
 62    * The background color of this renderer.
 63    *
 64    * @type {!string}
 65    * @protected
 66    */
 67   this._backgroundColor = '#000000';
 68 
 69   /**
 70    * The HTML container of this renderer, E.g a name of a <div>.
 71    *
 72    * @type {?Element}
 73    * @protected
 74    */
 75   this._container = null;
 76 
 77   /**
 78    * The Canvas of this renderer.
 79    *
 80    * @type {?Element}
 81    * @protected
 82    */
 83   this._canvas = null;
 84 
 85   /**
 86    * The WebGL context of this renderer.
 87    *
 88    * @type {?Object}
 89    * @protected
 90    */
 91   this._gl = null;
 92 
 93 };
 94 // inherit from X.base
 95 goog.inherits(X.renderer, X.base);

125  *
126  * @param {!number} width The width for this renderer.
127  */
128 X.renderer.prototype.setWidth = function(width) {
129 
130   if (this._canvas) {
131 
132     // the canvas was already created, let's update it
133     this._canvas.style.setProperty('width', width.toString());
134 
135   }
136 
137   this._width = width;
138 
139 };

205 /**
206  * Get the container of this renderer.
207  *
208  * @return {!Element} The container of this renderer as a DOM object.
209  * @throws {X.exception} An exception if the <body> could not be found.
210  */
211 X.renderer.prototype.getContainer = function() {
212 
213   // if no _container is associated, use the document.body
214   if (!this._container) {
215 
216     var _document = goog.dom.getDocument();
217     var body = _document.body;
218 
219     if (!body) {
220 
221       // throw exception when we can not find the body
222       throw new X.exception('Fatal: Could not find <body></body>!');
223 
224     }
225 
226     this._container = body;
227 
228   }
229 
230   // return the _container
231   return this._container;
232 
233 };

243 
244   if (!container) {
245 
246     // throw exception if the container is invalid
247     throw new X.exception('Fatal: Could not find container!');
248 
249   }
250 
251   this._container = container;
252 
253 };

257  * Set the container for this renderer using an id of a DOM object.

259  * @param {!string} containerId An id of a DOM object.

261 X.renderer.prototype.setContainerById = function(containerId) {
262 
263   // retrieve the DOM object with the given id
264   var container = goog.dom.getElement(containerId);
265 
266   // try to set it as a container
267   this.setContainer(container);

272 /**
273  * Create the canvas of this renderer inside the configured container and using
274  * attributes like width, height, backgroundColor etc. Then, initialize the
275  * WebGL context. All this will only happen once, no matter how often this
276  * method is called.
277  *
278  * @throws {X.exception} An exception if there were problems during WebGL
279  *           initialization.
280  */
281 X.renderer.prototype.init = function() {
282 
283   // if the canvas already exists, exit now
284   if (this._canvas) {
285     return;
286   }
287 
288   // create a canvas object with certain properties
289   var canvas = goog.dom.createDom('canvas');
290   canvas.style.setProperty('width', this.getWidth().toString());
291   canvas.style.setProperty('height', this.getHeight().toString());
292   canvas.style.setProperty('background-color', this.getBackgroundColor()
293       .toString());
294 
295   // append it to the container
296   goog.dom.appendChild(this.getContainer(), canvas);
297 
298   // WebGL initialization
299 
300   //
301   // Step1: Get Context of canvas
302   //
303   try {
304 
305     var gl = canvas.getContext('experimental-webgl');
306     // TODO do we need 2d canvas in a 2d case??
307     // gl = canvas.getContext('2d');
308 
309   } catch (e) {
310 
311     throw new X.exception('Fatal: Exception while getting GL Context!\n' + e);
312 
313   }
314 
315   //
316   // Step2: Check if we got the context, if not, WebGL is not supported
317   //
318   if (!gl) {
319 
320     throw new X.exception('Fatal: WebGL not supported!');
321 
322   }
323 
324   //
325   // Step3: Configure the context
326   //
327   try {
328 
329     gl.viewport(0, 0, this.getWidth(), this.getHeight());
330 
331   } catch (e) {
332 
333     throw new X.exception('Fatal: Exception while accessing GL Context!\n' + e);
334 
335   }
336 
337   // WebGL initialization done
338 
339   this._gl = gl;
340   this._canvas = canvas;
341 
342 };
