 30 // provides
 31 goog.provide('X.exception');
 32 
 33 // requires
 34 goog.require('X.base');
 35 
 36 
 37 
 38 /**
 39  * The general XTK exception.
 40  * 
 41  * @param {?string} message An error message.
 42  * @constructor
 43  * @extends X.base
 44  */
 45 X.exception = function(message) {
 46 
 47   var validMessage = 'Unknown error!';
 48   
 49   if (goog.isDefAndNotNull(message)) {
 50     
 51     // use the given message
 52     validMessage = message;
 53     
 54   }
 55   
 56   //
 57   // call the standard constructor of X.base
 58   goog.base(this);
 59   
 60   //
 61   // class attributes
 62   
 63   /**
 64    * @inheritDoc
 65    * @const
 66    */
 67   this._className = 'exception';
 68   
 69   /**
 70    * The message of this exception.
 71    * 
 72    * @type {!string}
 73    * @protected
 74    */
 75   this._message = 'Fatal: ' + validMessage;
 76   
 77   /**
 78    * The associated error object of this exception.
 79    * 
 80    * @type {Error}
 81    * @private
 82    */
 83   var _err = new Error();
 84   
 85   /**
 86    * The stack trace of this exception.
 87    * 
 88    * @type {!string}
 89    * @private
 90    */
 91   this._stackTrace_ = _err.stack;
 92   
 93 };
 94 goog.inherits(X.exception, X.base);
 95 
 96 // export symbols (required for advanced compilation)
 97 goog.exportSymbol('X.exception', X.exception);
