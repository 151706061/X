 40 /**
 41  * The superclass for all X.base-objects. All derived objects will be registered
 42  * for event handling.
 43  * 
 44  * @constructor
 45  * @extends goog.events.EventTarget
 46  */
 47 X.base = function() {
 48 
 49   //
 50   // register this class within the event system by calling the superclass
 51   // constructor
 52   goog.base(this);
 53   
 54   //
 55   // class attributes
 56   
 57   /**
 58    * The className of this class.
 59    * 
 60    * @type {string}
 61    * @protected
 62    */
 63   this._className = 'base';
 64   
 65   /**
 66    * The 'dirty' flag of this object.
 67    * 
 68    * @type {boolean}
 69    * @protected
 70    */
 71   this._dirty = false;
 72 };
 73 // enable events
 74 goog.inherits(X.base, goog.events.EventTarget);

 77 /**
 78  * Print the className of the current X.base-object.
 79  * 
 80  * @return {string} The className of the current X.object.
 81  */

101 /**
102  * Print the className and all attributes of the current X.base-object.
103  * 
104  * @return {string} A string representation of the current X.base-object.
105  */
106 X.base.prototype.print = function() {
107 
108   var attributes = Object.keys(this);
109   var attributesStringList = '';
110   var className = '<unknown>';
111   
112   // loop through the attributes of a class
113   var a = 0;
114   for (a in attributes) {
115     
116     var aName = 'this.' + attributes[a];
117     var aValue = eval(aName);
118     
119     // catch the className, since we want to display it differently
120     if (aName == 'this._className') {
121       className = aValue;
122     } else {
123       attributesStringList += aName + ': ' + aValue + '\n';
124     }
125     
126   }
127   
128   // build string output
129   var output = '== X.' + className + ' ==\n';
130   output += attributesStringList;
131   
132   return output;
133   
134 };
