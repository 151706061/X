 39 /**
 40  * Create a container for a scalar array.
 41  * 
 42  * @constructor
 43  * @extends X.base
 44  */
 45 X.scalars = function() {
 46 
 47   //
 48   // call the standard constructor of X.base
 49   goog.base(this);
 50   
 51   //
 52   // class attributes
 53   
 54   /**
 55    * @inheritDoc
 56    * @const
 57    */
 58   this['_className'] = 'scalars';
 59   
 60   // the global id counter
 61   var counter = window["X.Counter"];
 62   // ..get a new unique id
 63   counter.increment();
 64   
 65   /**
 66    * The uniqueId of this object. Each object in XTK has a uniqueId.
 67    * 
 68    * @type {number}
 69    * @protected
 70    */
 71   this['_id'] = counter.value();
 72   
 73   /**
 74    * The file containing the scalars.
 75    * 
 76    * @type {?X.file|string}
 77    * @protected
 78    */
 79   this._file = null;
 80   
 81   /**
 82    * The array containing the scalars.
 83    * 
 84    * @type {?Array}
 85    * @protected
 86    */
 87   this._array = null;
 88   
 89   /**
 90    * The WebGL-ready version of the scalars array.
 91    * 
 92    * @type {?Array}
 93    * @protected
 94    */
 95   this._glArray = null;
 96   
 97   /**
 98    * The min. scalar.
 99    * 
100    * @type {number}
101    * @protected
102    */
103   this._min = Infinity;
104   
105   /**
106    * The max. scalar.
107    * 
108    * @type {number}
109    * @protected
110    */
111   this._max = -Infinity;
112   
113   /**
114    * The color to map the min. scalar.
115    * 
116    * @type {!Array}
117    * @protected
118    */
119   this['_minColor'] = [0, 1, 0];
120   
121   /**
122    * The color to map the max. scalar.
123    * 
124    * @type {!Array}
125    * @protected
126    */
127   this['_maxColor'] = [1, 0, 0];
128   
129   /**
130    * The lower threshold.
131    * 
132    * @type {number}
133    * @protected
134    */
135   this['_minThreshold'] = -Infinity;
136   
137   /**
138    * The upper threshold.
139    * 
140    * @type {number}
141    * @protected
142    */
143   this['_maxThreshold'] = Infinity;
144   
145 };
146 // inherit from X.base
147 goog.inherits(X.scalars, X.base);

163  * Get the file of this scalars container.
164  * 
165  * @return {?X.file|string} The file of this scalars container.
166  */
167 X.scalars.prototype.file = function() {
168 
169   return this._file;
170   
171 };
172 
173 
174 /**
175  * Set the file for this scalars container.
176  * 
177  * @param {?X.file|string} file The file path or an X.file object containing the
178  *          path.
179  */
180 X.scalars.prototype.setFile = function(file) {
181 
182   if (!goog.isDefAndNotNull(file)) {
183     
184     // null files are allowed
185     this._file = null;
186     return;
187     
188   }
189   
190   if (goog.isString(file)) {
191     
192     file = new X.file(file);
193     
194   }
195   
196   this._file = file;
197   
198 };

201 /**
202  * Set the WebGL-ready array containing the scalars. This marks this object as
203  * dirty so the X.renderer can pick it up. Should be only used internally.
204  * 
205  * @param {Array} array The WebGL-ready array.
206  */
207 X.scalars.prototype.setGlArray = function(array) {
208 
209   this._glArray = array;
210   
211   // also, mark this dirty so the renderer can pick it up
212   this._dirty = true;
213   
214 };

217 /**
218  * Set the array containing the scalars. This array has to be WebGL-ready
219  * meaning that it has to match X.object.points().length(), which equals 3
220  * entries for each vertex (X,Y,Z coordinates). Calling this method marks this
221  * object as dirty so the X.renderer can pick it up. This method should be used
222  * externally.
223  * 
224  * @param {Array} array The WebGL-ready array matching
225  *          X.object.points().length() in size.
226  */
227 X.scalars.prototype.setArray = function(array) {
228 
229   this._array = array;
230   this._glArray = array;
231   
232   // also, mark this dirty so the renderer can pick it up
233   this._dirty = true;
234   
235 };

262 /**
263  * Set the color range to linear map the scalars to colors.
264  * 
265  * @param {Array} minColor The color corresponding to the min. scalar value.
266  * @param {Array} maxColor The color corresponding to the max. scalar value.
267  * @throws {Error} An exception if the given colors are invalid.
268  */
269 X.scalars.prototype.setColorRange = function(minColor, maxColor) {
270 
271   if (!goog.isDefAndNotNull(minColor) || !(minColor instanceof Array) ||
272       (minColor.length != 3)) {
273     
274     throw new Error('Invalid min. color.');
275     
276   }
277   
278   if (!goog.isDefAndNotNull(maxColor) || !(maxColor instanceof Array) ||
279       (maxColor.length != 3)) {
280     
281     throw new Error('Invalid max. color.');
282     
283   }
284   
285   this['_minColor'] = minColor;
286   this['_maxColor'] = maxColor;
287   
288 };
289 
290 
291 /**
292  * Get the color range which maps the scalars to colors in a linear fashion.
293  * 
294  * @return {Array} An array of length 2 where the min. color is element 0 and
295  *         max. color is element 1. The elements are again arrays with length 3.
296  */
297 X.scalars.prototype.colorRange = function() {
298 
299   return [this['_minColor'], this['_maxColor']];
300   
301 };
