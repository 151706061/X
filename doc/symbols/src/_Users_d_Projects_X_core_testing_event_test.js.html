 46 function testXeventPanEvent() {
 47

 49   var pe = new X.event.PanEvent();
 50   var pe2 = new X.event.PanEvent();
 51   
 52   // .. should be always the same type
 53   assertEquals(pe.type, pe2.type);
 54

 56   var _x = 1;

 59   
 60   // check the distances..
 61   
 62   // the default distance should always be 0,0
 63   assertEquals(pe2._distance.x, 0);
 64   assertEquals(pe2._distance.y, 0);
 65   
 66   // .. test for the custom distance
 67   assertEquals(pe._distance.x, _x);
 68   assertEquals(pe._distance.y, _y);
 69   
 70 }
