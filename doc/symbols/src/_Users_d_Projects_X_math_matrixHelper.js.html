 19 /**
 20  * Create a flattened, one-dimensional representation of a goog.math.Matrix.
 21  *
 22  * @this {goog.math.Matrix}
 23  * @return {Array} A one-dimensional representation of this goog.math.Matrix.
 24  */
 25 X.matrixHelper.flatten = function() {
 26 
 27   var result = [];
 28 
 29   var dimensions = this.getSize();
 30 
 31   if (dimensions.height == 0 || dimensions.width == 0) {
 32     return [];
 33   }
 34 
 35   var i, j;
 36   for (j = 0; j < dimensions.height; j++) {
 37     for (i = 0; i < dimensions.width; i++) {
 38       result.push(this.getValueAt(i, j));
 39     }
 40   }
 41   return result;
 42 
 43 };

 47  * register the function to the goog.math.Matrix API
 48  */
 49 goog.math.Matrix.prototype.flatten = X.matrixHelper.flatten;

 52 /**
 53  * Translate a 3x3 or 4x4 goog.math.Matrix by a vector. In the 3x3 case, the
 54  * vector has to be 2 dimensional. In the 4x4 case, the vector has to be 3
 55  * dimensional.
 56  *
 57  * @this {goog.math.Matrix}
 58  * @param {!goog.math.Vec2|!goog.math.Vec3} vector The translation vector.
 59  * @return {goog.math.Matrix} The result of this translation.
 60  * @throws {X.exception} An exception if the translation fails.
 61  */
 62 X.matrixHelper.translate = function translate(vector) {
 63 
 64   if (!this.isSquare()) {
 65 
 66     throw new X.exception('Fatal: Can not translate non-square matrix!');
 67 
 68   }
 69 
 70   var dimensions = this.getSize();
 71 
 72   var transformationMatrix = goog.math.Matrix
 73       .createIdentityMatrix(dimensions.height);
 74 
 75   if (vector instanceof goog.math.Vec2 && dimensions.height == 3) {
 76 
 77     transformationMatrix.setValueAt(0, 2, vector.x);
 78     transformationMatrix.setValueAt(1, 2, vector.y);
 79 
 80   } else if (vector instanceof goog.math.Vec3 && dimensions.height == 4) {
 81 
 82     transformationMatrix.setValueAt(0, 3, vector.x);
 83     transformationMatrix.setValueAt(1, 3, vector.y);
 84     transformationMatrix.setValueAt(2, 3, vector.z);
 85 
 86   } else {
 87 
 88     throw new X.exception('Fatal: Translation failed!');
 89 
 90   }
 91 
 92   // now multiply this matrix with the transformationMatrix and return it..
 93   return this.multiply(transformationMatrix);
 94 
 95 };

101 goog.math.Matrix.prototype.translate = X.matrixHelper.translate;

104 /**
105  * Rotate a 4x4 goog.math.Matrix by an angle around an axis.
106  *
107  * @this {goog.math.Matrix}
108  * @param {!number} angle The rotation angle.
109  * @param {!goog.math.Vec3} iaxis The axis to rotate around.
110  * @return {goog.math.Matrix} The result of this rotation.
111  * @throws {X.exception} An exception if the rotation fails.
112  */
113 X.matrixHelper.rotate = function(angle, iaxis) {
114 
115   var dimensions = this.getSize();
116 
117   if (dimensions.height != 4 || !this.isSquare()) {
118 
119     throw new X.exception('Fatal: Only 4x4 matrices can be rotated!');
120 
121   }
122 
123   if (!goog.isDefAndNotNull(iaxis) || !(iaxis instanceof goog.math.Vec3)) {
124 
125     throw new X.exception('Fatal: Invalid axis vector!');
126 
127   }
128 
129   if (!goog.isNumber(angle)) {
130 
131     throw new X.exception('Fatal: Invalid angle!');
132 
133   }
134 
135   // normalize the axis
136   var axis = iaxis.normalize();
137 
138   // trigonometrical fun
139   var cos = Math.cos(angle);
140   var sin = Math.sin(angle);
141 
142   var rotationMatrix = new goog.math.Matrix.createIdentityMatrix(4);
143 
144   // build rotation matrix according to
145   // http://en.wikipedia.org/wiki/Rotation_matrix#Rotation_matrix_from_axis_and_angle
146 
147   rotationMatrix.setValueAt(0, 0, cos + axis.x * axis.x * (1 - cos));
148   rotationMatrix.setValueAt(0, 1, axis.x * axis.y * (1 - cos) - axis.z * sin);
149   rotationMatrix.setValueAt(0, 2, axis.x * axis.z * (1 - cos) + axis.y * sin);
150 
151   rotationMatrix.setValueAt(1, 0, axis.y * axis.x * (1 - cos) + axis.z * sin);
152   rotationMatrix.setValueAt(1, 1, cos + axis.y * axis.y * (1 - cos));
153   rotationMatrix.setValueAt(1, 2, axis.y * axis.z * (1 - cos) - axis.x * sin);
154 
155   rotationMatrix.setValueAt(2, 0, axis.z * axis.x * (1 - cos) - axis.y * sin);
156   rotationMatrix.setValueAt(2, 1, axis.z * axis.y * (1 - cos) + axis.x * sin);
157   rotationMatrix.setValueAt(2, 2, cos + axis.z * axis.z * (1 - cos));
158 
159   // now multiply and return
160   return this.multiply(rotationMatrix);
161 
162 };

171 /**
172  * Multiply 3x3 or 4x4 goog.math.Matrix by a vector. In the 3x3 case, the vector
173  * has to be at least 2 dimensional. In the 4x4 case, the vector has to be at
174  * least 3 dimensional.
175  *
176  * @this {goog.math.Matrix}
177  * @param {!goog.math.Vec2|!goog.math.Vec3} vector The multiplication vector.
178  * @return {!goog.math.Vec2|!goog.math.Vec3} The result of this multiplication.
179  * @throws {X.exception} An exception if the multiplication fails.
180  */
181 X.matrixHelper.multiplyByVector = function(vector) {
182 
183   var dimensions = this.getSize();
184 
185   // we need to convert the vector to a matrix
186   //
187   var vectorAsArray = new Array(dimensions.width);
188 
189   // set all values to one TODO: is zero better?
190   var i;
191   for (i = 0; i < vectorAsArray.length; i++) {
192 
193     vectorAsArray[i] = new Array(1);
194     vectorAsArray[i][0] = 1;
195 
196   }
197 
198   if (vector instanceof goog.math.Vec2 && dimensions.width >= 2) {
199 
200     vectorAsArray[0][0] = vector.x;
201     vectorAsArray[1][0] = vector.y;
202 
203   } else if (vector instanceof goog.math.Vec3 && dimensions.width >= 3) {
204 
205     vectorAsArray[0][0] = vector.x;
206     vectorAsArray[1][0] = vector.y;
207     vectorAsArray[2][0] = vector.z;
208 
209   } else {
210 
211     throw new X.exception('Fatal: Multiplication by vector failed!');
212 
213   }
214 
215   // now convert the vectorAsArray to a matrix and multiply
216   var vectorAsMatrix = new goog.math.Matrix(vectorAsArray);
217 
218   // ...and multiply it
219   var resultMatrix = this.multiply(vectorAsMatrix);
220   var resultDimensions = resultMatrix.getSize();
221 
222   var resultVector;
223 
224   // .. convert to vector and return it
225   if (resultDimensions.height >= 3) {
226 
227     // 3d vector
228     resultVector = new goog.math.Vec3(resultMatrix.getValueAt(0, 0),
229         resultMatrix.getValueAt(1, 0), resultMatrix.getValueAt(2, 0));
230 
231   } else {
232 
233     // 2d vector
234     resultVector = new goog.math.Vec2(resultMatrix.getValueAt(0, 0),
235         resultMatrix.getValueAt(1, 0));
236 
237   }
238 
239   return resultVector;
240 
241 };
