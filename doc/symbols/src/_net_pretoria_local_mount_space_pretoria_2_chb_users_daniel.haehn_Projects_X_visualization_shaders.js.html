 38 /**
 39  * Create a pair of shaders which consists of a vertex and a fragment shader.
 40  * 
 41  * @constructor
 42  * @extends X.base
 43  */
 44 X.shaders = function() {
 45 
 46   //
 47   // call the standard constructor of X.base
 48   goog.base(this);
 49   
 50   //
 51   // class attributes
 52   
 53   /**
 54    * @inheritDoc
 55    * @const
 56    */
 57   this['_className'] = 'shader';
 58   
 59   /**
 60    * The vertex shader source of this shader pair. By default, a basic shader
 61    * supporting vertex positions and vertex colors is defined.
 62    * 
 63    * @type {!string}
 64    * @protected
 65    */
 66   this._vertexshaderSource = '';
 67   var t = '';
 68   t += 'precision mediump float;\n';
 69   t += '\n';
 70   t += 'attribute vec3 vertexPosition;\n';
 71   t += 'attribute vec3 vertexNormal;\n';
 72   t += 'attribute vec3 vertexColor;\n';
 73   t += 'attribute vec2 vertexTexturePos;\n';
 74   t += '\n';
 75   t += 'uniform mat4 view;\n';
 76   t += 'uniform mat4 perspective;\n';
 77   t += 'uniform vec3 center;\n';
 78   t += 'uniform mat4 objectTransform;\n';
 79   t += 'uniform bool useObjectColor;\n';
 80   t += 'uniform vec3 objectColor;\n';
 81   t += 'uniform float pointSize;\n';
 82   t += '\n';
 83   t += 'varying vec4 fVertexPosition;\n';
 84   t += 'varying vec3 fragmentColor;\n';
 85   t += 'varying vec2 fragmentTexturePos;\n';
 86   t += 'varying vec3 fTransformedVertexNormal;\n';
 87   t += '\n';
 88   t += 'void main(void) {\n';
 89   // setup varying -> fragment shader
 90   // use the old mat3 constructor to be compatible with mac/safari
 91   t += '  fTransformedVertexNormal = mat3(view[0].xyz,view[1].xyz,view[2].xyz) * ';
 92   t += 'mat3(objectTransform[0].xyz,objectTransform[1].xyz,objectTransform[2].xyz) * ';
 93   t += 'vertexNormal;\n';
 94   // t += ' vec4 gVertexPosition = vec4(fVertexPosition.xyz - focus, 1.0);\n';
 95   t += '  vec3 vertexPosition2 = vertexPosition - center;\n';
 96   t += '  fVertexPosition = view * objectTransform * vec4(vertexPosition2, 1.0);\n';
 97   t += '  fragmentTexturePos = vertexTexturePos;\n';
 98   t += '  if (useObjectColor) {\n';
 99   t += '    fragmentColor = objectColor;\n';
100   t += '  } else {\n';
101   t += '    fragmentColor = vertexColor;\n';
102   t += '  }\n';
103   // setup vertex Point Size and Position in the GL context
104   t += '  gl_PointSize = pointSize;\n';
105   t += '  gl_Position = perspective * fVertexPosition;\n';
106   t += '}\n';
107   this._vertexshaderSource = t;
108   
109   /**
110    * The fragment shader source of this shader pair. By default, a basic shader
111    * supporting fragment colors is defined.
112    * 
113    * @type {!string}
114    * @protected
115    */
116   this._fragmentshaderSource = '';
117   var t2 = '';
118   // android only guarantees medium precision
119   t2 += 'precision mediump float;\n';
120   t2 += '\n';
121   t2 += 'uniform bool usePicking;\n';
122   t2 += 'uniform bool useTexture;\n';
123   t2 += 'uniform sampler2D textureSampler;\n';
124   t2 += 'uniform float objectOpacity;\n';
125   t2 += 'uniform float volumeLowerThreshold;\n';
126   t2 += 'uniform float volumeUpperThreshold;\n';
127   t2 += 'uniform float volumeScalarMin;\n';
128   t2 += 'uniform float volumeScalarMax;\n';
129   t2 += '\n';
130   t2 += 'varying vec4 fVertexPosition;\n';
131   t2 += 'varying vec3 fragmentColor;\n';
132   t2 += 'varying vec2 fragmentTexturePos;\n';
133   t2 += 'varying vec3 fTransformedVertexNormal;\n';
134   t2 += '\n';
135   t2 += 'void main(void) {\n';
136   // in picking mode, we don't want any extras but just the plain color
137   t2 += ' if (usePicking) {\n';
138   t2 += '   gl_FragColor = vec4(fragmentColor, 1.0);\n';
139   t2 += ' } else if (useTexture) {\n';
140   t2 += '   gl_FragColor = texture2D(textureSampler,';
141   t2 += '   vec2(fragmentTexturePos.s,fragmentTexturePos.t));\n';
142   // threshold functionality for 1-channel volumes
143   t2 += '   float _volumeLowerThreshold = (volumeLowerThreshold / volumeScalarMax);\n';
144   t2 += '   float _volumeUpperThreshold = (volumeUpperThreshold / volumeScalarMax);\n';
145   t2 += '   if (gl_FragColor.r < _volumeLowerThreshold ||\n';
146   t2 += '       gl_FragColor.r > _volumeUpperThreshold) {\n';
147   t2 += '     discard;\n';
148   t2 += '   };\n';
149   t2 += ' gl_FragColor.a = objectOpacity;\n';
150   t2 += ' } else {\n';
151   // configure advanced lighting
152   t2 += '   vec3 nNormal = normalize(fTransformedVertexNormal);\n';
153   t2 += '   vec3 light = vec3(0.0, 0.0, 1.0);\n';
154   // t2 += ' vec3 lightDirection = vec3(-10.0, 4.0, -20.0);\n';
155   // I liked the following better
156   t2 += '   vec3 lightDirection = vec3(0,0,-10);\n';
157   t2 += '   lightDirection = normalize(lightDirection);\n';
158   t2 += '   vec3 eyeDirection = normalize(-fVertexPosition.xyz);\n';
159   t2 += '   vec3 reflectionDirection = reflect(-lightDirection, nNormal);\n';
160   // t2 += ' vec3 reflectionDirection = nNormal;\n'; <-- to disable reflection
161   // configure specular (10.0 is material property), diffuse and ambient
162   t2 += '   float specular = pow(max(dot(reflectionDirection, eyeDirection), 0.0), 10.0);\n';
163   t2 += '   float diffuse = 0.8 * max(dot(nNormal, light), 0.0);\n';
164   t2 += '   float ambient = 0.3;\n';
165   // .. and now setup the fragment color using these three values and the
166   // opacity
167   t2 += '   gl_FragColor = vec4(fragmentColor * ambient +\n';
168   t2 += '                       fragmentColor * diffuse +\n';
169   t2 += '                       vec3(0.2, 0.2, 0.2) * specular,\n';
170   t2 += '                       objectOpacity);\n';
171   t2 += ' }\n';
172   t2 += '}\n';
173   this._fragmentshaderSource = t2;
174   
175 
176 };
177 // inherit from X.base
178 goog.inherits(X.shaders, X.base);

245 /**
246  * Checks if this configured shaders object is valid in terms of using the
247  * defined attributes and uniforms.
248  * 
249  * @return {boolean} TRUE or FALSE depending on success.
250  * @throws {Error} An exception if the shader is invalid.
251  */
252 X.shaders.prototype.validate = function() {
253 
254   // check if the shader sources are compatible to the attributes and uniforms
255   
256   var attributes = Object.keys(X.shaders.attributes);
257   var uniforms = Object.keys(X.shaders.uniforms);
258   
259   // check if all attributes are used either in the vertex or the fragment
260   // shader
261   var attributesValid = attributes.every(function(a) {
262 
263     a = eval("X.shaders.attributes." + a);
264     return (this._vertexshaderSource.search(a) != -1) ||
265         (this._fragmentshaderSource.search(a) != -1);
266     
267   }.bind(this));
268   
269   if (!attributesValid) {
270     
271     throw new Error('Could not find all attributes in the shader sources.');
272     
273   }
274   
275   // check if all attributes are used either in the vertex or the fragment
276   // shader
277   var uniformsValid = uniforms.every(function(u) {
278 
279     u = eval("X.shaders.uniforms." + u);
280     return (this._vertexshaderSource.search(u) != -1) ||
281         (this._fragmentshaderSource.search(u) != -1);
282     
283   }.bind(this));
284   
285   if (!uniformsValid) {
286     
287     throw new Error('Could not find all uniforms in the shader sources.');
288     
289   }
290   
291   return true;
292   
293 };
