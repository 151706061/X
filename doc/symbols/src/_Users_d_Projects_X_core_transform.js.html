 16 /**
 17  * Create a transform.
 18  * 
 19  * @constructor
 20  * @extends X.base
 21  */
 22 X.transform = function() {
 23 
 24   //
 25   // call the standard constructor of X.base
 26   goog.base(this);
 27   
 28   //
 29   // class attributes
 30   
 31   /**
 32    * @inheritDoc
 33    * @const
 34    */
 35   this._className = 'transform';
 36   
 37   /**
 38    * The transformation matrix.
 39    * 
 40    * @type {!X.matrix}
 41    * @protected
 42    */
 43   this._matrix = X.matrix.createIdentityMatrix(4);
 44   
 45   /**
 46    * The transformation matrix as a 'ready-to-use'-gl version.
 47    * 
 48    * @type {!Object}
 49    * @protected
 50    */
 51   this._glMatrix = new Float32Array(this._matrix.flatten());
 52   
 53 };
 54 // inherit from X.base
 55 goog.inherits(X.transform, X.base);

 58 /**
 59  * Rotate around the X-axis.
 60  * 
 61  * @param {number} angle The angle to rotate in degrees.
 62  * @throws {X.exception} An exception, if the given angle is invalid.
 63  */
 64 X.transform.prototype.rotateX = function(angle) {
 65 
 66   if (!goog.isNumber(angle) || angle < -360 || angle > 360) {
 67     
 68     throw new X.exception('Fatal: Invalid angle!');
 69     
 70   }
 71   
 72   var angleInRadii = angle * Math.PI / 180;
 73   
 74   this._matrix = this._matrix.rotate(angleInRadii, new goog.math.Vec3(0, 1, 0));
 75   this._glMatrix = new Float32Array(this._matrix.flatten());
 76   
 77 };

124 /**
125  * Translate on the X-axis.
126  * 
127  * @param {number} distance The distance to move.
128  * @throws {X.exception} An exception, if the given distance is invalid.
129  */
130 X.transform.prototype.translateX = function(distance) {
131 
132   if (!goog.isNumber(distance)) {
133     
134     throw new X.exception('Fatal: Invalid distance!');
135     
136   }
137   
138   var vector = new goog.math.Vec3(distance, 0, 0);
139   
140   this._matrix = this._matrix.translate(vector);
141   this._glMatrix = new Float32Array(this._matrix.flatten());
142   
143 };
