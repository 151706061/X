 39 /**
 40  * Create a displayable volume which consists of X.slices in X,Y and Z direction
 41  * and can also be volume rendered.
 42  * 
 43  * @constructor
 44  * @extends X.object
 45  */
 46 X.volume = function() {
 47 
 48   //
 49   // call the standard constructor of X.base
 50   goog.base(this);
 51   
 52   //
 53   // class attributes
 54   
 55   /**
 56    * @inheritDoc
 57    * @const
 58    */
 59   this['_className'] = 'volume';
 60   
 61   /**
 62    * The center of this volume.
 63    * 
 64    * @type {!Array}
 65    * @protected
 66    */
 67   this._center = [0, 0, 0];
 68   
 69   /**
 70    * The dimensions of this volume.
 71    * 
 72    * @type {!Array}
 73    * @protected
 74    */
 75   this._dimensions = [10, 10, 10];
 76   
 77   /**
 78    * The spacing of this volume.
 79    * 
 80    * @type {!Array}
 81    * @protected
 82    */
 83   this._spacing = [1, 1, 1];
 84   
 85   /**
 86    * The index of the currently shown slice in X-direction.
 87    * 
 88    * @type {!number}
 89    * @public
 90    */
 91   this['_indexX'] = 0;
 92   
 93   /**
 94    * The index of the formerly shown slice in X-direction.
 95    * 
 96    * @type {!number}
 97    * @protected
 98    */
 99   this._indexXold = 0;
100   
101   /**
102    * The index of the currently shown slice in Y-direction.
103    * 
104    * @type {!number}
105    * @public
106    */
107   this['_indexY'] = 0;
108   
109   /**
110    * The index of the formerly shown slice in Y-direction.
111    * 
112    * @type {!number}
113    * @protected
114    */
115   this._indexYold = 0;
116   
117   /**
118    * The index of the currently shown slice in Z-direction.
119    * 
120    * @type {!number}
121    * @public
122    */
123   this['_indexZ'] = 0;
124   
125   /**
126    * The index of the formerly shown slice in Z-direction.
127    * 
128    * @type {!number}
129    * @protected
130    */
131   this._indexZold = 0;
132   
133   /**
134    * The X.object holding the slices in X-direction.
135    * 
136    * @type {!X.object}
137    * @protected
138    */
139   this._slicesX = new X.object();
140   
141   /**
142    * The X.object holding the slices in Y-direction.
143    * 
144    * @type {!X.object}
145    * @protected
146    */
147   this._slicesY = new X.object();
148   
149   /**
150    * The X.object holding the slices in Z-direction.
151    * 
152    * @type {!X.object}
153    * @protected
154    */
155   this._slicesZ = new X.object();
156   
157   /**
158    * The upper threshold for this volume.
159    * 
160    * @type {number}
161    * @public
162    */
163   this['_lowerThreshold'] = 0;
164   
165   /**
166    * The upper threshold for this volume.
167    * 
168    * @type {number}
169    * @public
170    */
171   this['_upperThreshold'] = 1000;
172   
173   /**
174    * The scalar range of this volume.
175    * 
176    * @type {!Array}
177    * @protected
178    */
179   this._scalarRange = [0, 1000];
180   
181   /**
182    * The toggle for volume rendering or cross-sectional slicing.
183    * 
184    * @type {boolean}
185    * @public
186    */
187   this['_volumeRendering'] = false;
188   this._volumeRenderingOld = false;
189   
190 
191   /**
192    * The direction for the volume rendering. This is used for caching.
193    * 
194    * @type {!number}
195    * @private
196    */
197   this._volumeRenderingDirection = 0;
198   
199 };

204 /**
205  * Create the volume.
206  * 
207  * @private
208  */
209 X.volume.prototype.create_ = function() {
210 
211   // remove all old children
212   this.children().length = 0;
213   
214   // add the new children
215   this.children().push(this._slicesX);
216   this.children().push(this._slicesY);
217   this.children().push(this._slicesZ);
218   
219   //
220   // create the slices
221   var xyz = 0; // 0 for x, 1 for y, 2 for z
222   for (xyz = 0; xyz < 3; xyz++) {
223     
224     var halfDimension = (this._dimensions[xyz] - 1) / 2;
225     
226     var _indexCenter = halfDimension;
227     
228     var i = 0;
229     for (i = 0; i < this._dimensions[xyz]; i++) {
230       
231       var _position = (-halfDimension * this._spacing[xyz]) +
232           (i * this._spacing[xyz]);
233       
234       var _center = new Array([this._center[0] + _position, this._center[1],
235                                this._center[2]], [this._center[0],
236                                                   this._center[1] + _position,
237                                                   this._center[2]],
238           [this._center[0], this._center[1], this._center[2] + _position]);
239       
240       var _front = new Array([1, 0, 0], [0, 1, 0], [0, 0, 1]);
241       var _up = new Array([0, 1, 0], [0, 0, -1], [0, 1, 0]);
242       
243       // the container and indices
244       var slices = this.children()[xyz].children();
245       
246       // dimensions
247       var width = 0;
248       var height = 0;
249       if (xyz == 0) {
250         // for x slices
251         width = this._dimensions[2] * this._spacing[2] - this._spacing[2];
252         height = this._dimensions[1] * this._spacing[1] - this._spacing[1];
253       } else if (xyz == 1) {
254         // for y slices
255         width = this._dimensions[0] * this._spacing[0] - this._spacing[0];
256         height = this._dimensions[2] * this._spacing[2] - this._spacing[2];
257       } else if (xyz == 2) {
258         width = this._dimensions[0] * this._spacing[0] - this._spacing[0];
259         height = this._dimensions[1] * this._spacing[1] - this._spacing[1];
260       }
261       
262       // .. new slice
263       var _slice = new X.slice(_center[xyz], _front[xyz], _up[xyz], width,
264           height);
265       _slice._volume = this;
266       
267       // only show the middle slice, hide everything else
268       _slice.setVisible(i == _indexCenter);
269       
270       // attach to all _slices with the correct slice index
271       slices.push(_slice);
272       
273     }
274     
275     // by default, all the 'middle' slices are shown
276     if (xyz == 0) {
277       this['_indexX'] = _indexCenter;
278       this._indexXold = _indexCenter;
279     } else if (xyz == 1) {
280       this['_indexY'] = _indexCenter;
281       this._indexYold = _indexCenter;
282     } else if (xyz == 2) {
283       this['_indexZ'] = _indexCenter;
284       this._indexZold = _indexCenter;
285     }
286   }
287   
288   this._dirty = true;
289   
290 };

292 /**
293  * Re-show the slices or re-activate the volume rendering for this volume.
294  * 
295  * @inheritDoc
296  */
297 X.volume.prototype.modified = function() {
298 
299   if (this['_volumeRendering'] != this._volumeRenderingOld) {
300     
301     // switch from slicing to volume rendering or vice versa
302     this._dirty = true;
303     this._volumeRenderingOld = this['_volumeRendering'];
304     
305   }
306   
307   if (this['_volumeRendering']) {
308     
309     // prepare volume rendering
310     this.volumeRendering_(this._volumeRenderingDirection);
311     
312   } else {
313     
314     // prepare slicing
315     this.slicing_();
316     
317   }
318   
319 
320   // call the superclass' modified method
321   X.volume.superClass_.modified.call(this);
322   
323 };
324 
325 /**
326  * Show the current slices which are set by this._indexX, this._indexY and
327  * this._indexZ and hide all others.
328  */
329 X.volume.prototype.slicing_ = function() {
330 
331   if (this._dirty) {
332     
333     // hide the volume rendering slices
334     var _child = this.children()[this._volumeRenderingDirection];
335     _child.setVisible(false);
336     
337   }
338   
339   // display the current slices in X,Y and Z direction
340   var xyz = 0; // 0 for x, 1 for y, 2 for z
341   for (xyz = 0; xyz < 3; xyz++) {
342     
343     var _child = this.children()[xyz];
344     var currentIndex = 0;
345     var oldIndex = 0;
346     
347     if (xyz == 0) {
348       currentIndex = this['_indexX'];
349       oldIndex = this._indexXold;
350       this._indexXold = this['_indexX'];
351     } else if (xyz == 1) {
352       currentIndex = this['_indexY'];
353       oldIndex = this._indexYold;
354       this._indexYold = this['_indexY'];
355     } else if (xyz == 2) {
356       currentIndex = this['_indexZ'];
357       oldIndex = this._indexZold;
358       this._indexZold = this['_indexZ'];
359     }
360     
361     // hide the old slice
362     _child.children()[parseInt(oldIndex, 10)].setVisible(false);
363     // show the current slice
364     _child.children()[parseInt(currentIndex, 10)].setVisible(true);
365     
366   }
367   
368 };

395 /**
396  * Threshold this volume. All pixel values smaller than lower or larger than
397  * upper are ignored during rendering.
398  * 
399  * @param {!number} lower The lower threshold value.
400  * @param {!number} upper The upper threshold value.
401  * @throws {Error} If the specified range is invalid.
402  */
403 X.volume.prototype.threshold = function(lower, upper) {
404 
405   if (!goog.isDefAndNotNull(lower) || !goog.isNumber(lower) ||
406       !goog.isDefAndNotNull(upper) || !goog.isNumber(upper) ||
407       (lower > upper) || (lower < this._scalarRange[0]) ||
408       (upper > this._scalarRange[1])) {
409     
410     throw new Error('Invalid threshold range.');
411     
412   }
413   
414   this['_lowerThreshold'] = lower;
415   this['_upperThreshold'] = upper;
416   
417 };

420 /**
421  * Toggle volume rendering or cross-sectional slicing of this X.volume.
422  * 
423  * @param {boolean} volumeRendering If TRUE, display volume rendering, if FALSE
424  *          display cross-sectional slices.

426 X.volume.prototype.setVolumeRendering = function(volumeRendering) {
427 
428   this._volumeRendering = volumeRendering;
429   
430 };

433 /**
434  * Set the center of this X.volume. This has to be called (for now) before a
435  * volume data gets loaded aka. before the first X.renderer.render() call.
436  * 
437  * @param {!Array} center The new center.
438  * @throws {Error} If the center is invalid.
439  */
440 X.volume.prototype.setCenter = function(center) {
441 
442   if (!goog.isDefAndNotNull(center) || !(center instanceof Array) ||
443       !(center.length == 3)) {
444     
445     throw new Error('Invalid center.');
446     
447   }
448   
449   this._center = center;
450   
451 };

454 /**
455  * Perform volume rendering of this volume along a specific direction. The
456  * direction is important since we show tiled 2d textures along the direction
457  * for a clean rendering experience.
458  * 
459  * @param {number} direction The direction of the volume rendering
460  *          (0==x,1==y,2==z).
461  */
462 X.volume.prototype.volumeRendering_ = function(direction) {
463 
464   if ((!this._dirty && !this['_volumeRendering']) ||
465       (!this._dirty && direction == this._volumeRenderingDirection)) {
466     
467     // we do not have to do anything
468     return;
469     
470   }
471   
472   // first, hide possible slicing slices
473   this.children()[0].children()[parseInt(this['_indexX'], 10)]
474       .setVisible(false);
475   this.children()[1].children()[parseInt(this['_indexY'], 10)]
476       .setVisible(false);
477   this.children()[2].children()[parseInt(this['_indexZ'], 10)]
478       .setVisible(false);
479   
480   // hide old volume rendering slices
481   var _child = this.children()[this._volumeRenderingDirection];
482   _child.setVisible(false);
483   
484   // show new volume rendering slices
485   _child = this.children()[direction];
486   _child.setVisible(true);
487   
488   // store the direction
489   this._volumeRenderingDirection = direction;
490   
491 };
