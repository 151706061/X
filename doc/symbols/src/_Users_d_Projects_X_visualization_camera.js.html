 20 /**
 21  * Create a camera.
 22  * 
 23  * @constructor
 24  * @param {number} width The width of the camera's viewport.
 25  * @param {number} height The height of the camera's viewport.
 26  * @extends X.base
 27  */
 28 X.camera = function(width, height) {
 29 
 30   // validate width and height
 31   if (!goog.isNumber(width) || !goog.isNumber(height)) {
 32     
 33     throw new X.exception(
 34         'Fatal: A camera needs valid width and height values.');
 35     
 36   }
 37   
 38   //
 39   // call the standard constructor of X.base
 40   goog.base(this);
 41   
 42   //
 43   // class attributes
 44   
 45   /**
 46    * @inheritDoc
 47    * @const
 48    */
 49   this._className = 'camera';
 50   
 51   /**
 52    * The field of view in degrees.
 53    * 
 54    * @type {number}
 55    * @const
 56    */
 57   this._fieldOfView = 45;
 58   
 59   /**
 60    * The position of this camera.
 61    * 
 62    * @type {goog.math.Vec3}
 63    * @protected
 64    */
 65   this._position = new goog.math.Vec3(0, 0, 100);
 66   
 67   /**
 68    * The focus point of this camera.
 69    * 
 70    * @type {goog.math.Vec3}
 71    * @protected
 72    */
 73   this._focus = new goog.math.Vec3(0, 0, 0);
 74   
 75   /**
 76    * The unit vector pointing to the top of the three-dimensional space.
 77    * 
 78    * @type {goog.math.Vec3}
 79    * @protected
 80    */
 81   this._up = new goog.math.Vec3(0, 1, 0);
 82   
 83   /**
 84    * The perspective matrix.
 85    * 
 86    * @type {X.matrix}
 87    * @protected
 88    */
 89   this._perspective = new Float32Array(this.calculatePerspective_(
 90       this._fieldOfView, (width / height), 1, 10000).flatten());
 91   
 92   /**
 93    * The view matrix.
 94    * 
 95    * @type {X.matrix}
 96    * @protected
 97    */
 98   this._view = this.lookAt_(this._position, this._focus);
 99   
100   /**
101    * The view matrix as a 'ready-to-use'-gl version.
102    * 
103    * @type {Object}
104    * @protected
105    */
106   this._glView = new Float32Array(this._view.flatten());
107   
108   /**
109    * The inverted and transposed view matrix as a 'ready-to-use'-gl version.
110    * 
111    * @type {Object}
112    * @protected
113    */
114   this._glViewInvertedTransposed = new Float32Array(new X.matrix(this._view
115       .getInverse().getTranspose()).flatten());
116   
117 };
118 // inherit from X.base
119 goog.inherits(X.camera, X.base);

122 /**
123  * Configures observes for a given interactor. The method sets up listeners for
124  * PAN, ROTATE and ZOOM events.
125  * 
126  * @param {!X.interactor} interactor The interactor which should be observed.
127  */
128 X.camera.prototype.observe = function(interactor) {
129 
130   if (!goog.isDefAndNotNull(interactor) ||
131       !(interactor instanceof X.interactor)) {
132     
133     throw new X.exception('Fatal: Could not observe the interactor.');
134     
135   }
136   
137   goog.events.listen(interactor, X.event.events.PAN, this.onPan.bind(this));
138   goog.events.listen(interactor, X.event.events.ROTATE, this.onRotate
139       .bind(this));
140   goog.events.listen(interactor, X.event.events.ZOOM, this.onZoom.bind(this));
141   
142 };

145 /**
146  * The callback for a ZOOM event.
147  * 
148  * @param {!X.event.ZoomEvent} event The event.
149  * @throws {X.exception} An exception if the event is invalid.
150  */
151 X.camera.prototype.onZoom = function(event) {
152 
153   if (!(event instanceof X.event.ZoomEvent)) {
154     
155     throw new X.exception('Fatal: Received no valid zoom event!');
156     
157   }
158   
159   if (event._in) {
160     
161     this.zoomIn(event._fast);
162     
163   } else {
164     
165     this.zoomOut(event._fast);
166     
167   }
168   
169 };

342 /**
343  * Calculate a perspective matrix based on the given values. This calculation is
344  * based on known principles of Computer Vision (Source: TODO?).
345  * 
346  * @param {number} fieldOfViewY The field of view in degrees in Y direction.
347  * @param {number} aspectRatio The aspect ratio between width and height of the
348  *          viewport.
349  * @param {number} zNearClippingPlane The Z coordinate of the near clipping
350  *          plane (close to the eye).
351  * @param {number} zFarClippingPlane The Z coordinate of the far clipping plane
352  *          (far from the eye).
353  * @return {X.matrix} The perspective matrix.
354  * @private
355  */
356 X.camera.prototype.calculatePerspective_ = function(fieldOfViewY, aspectRatio,
357     zNearClippingPlane, zFarClippingPlane) {
358 
359   var ymax = zNearClippingPlane * Math.tan(fieldOfViewY * Math.PI / 360.0);
360   var ymin = -ymax;
361   var xmin = ymin * aspectRatio;
362   var xmax = ymax * aspectRatio;
363   
364   return this.calculateViewingFrustum_(xmin, xmax, ymin, ymax,
365       zNearClippingPlane, zFarClippingPlane);
366   
367 };

370 /**
371  * Calculate the view frustum which is the three-dimensional area which is
372  * visible to the eye by 'trimming' the world space. This calculation is based
373  * on known principles of Computer Vision (Source:
374  * http://en.wikipedia.org/wiki/Viewing_frustum).
375  * 
376  * @param {number} left The Y coordinate of the left border.
377  * @param {number} right The Y coordinate of the right border.
378  * @param {number} bottom The X coordinate of the bottom border.
379  * @param {number} top The X coordinate of the top border.
380  * @param {number} znear The Z coordinate of the near the eye border.
381  * @param {number} zfar The Z coordinate of the far of the eye border.
382  * @return {X.matrix} The frustum matrix.
383  * @private
384  */
385 X.camera.prototype.calculateViewingFrustum_ = function(left, right, bottom,
386     top, znear, zfar) {
387 
388   var x = 2 * znear / (right - left);
389   var y = 2 * znear / (top - bottom);
390   var a = (right + left) / (right - left);
391   var b = (top + bottom) / (top - bottom);
392   var c = -(zfar + znear) / (zfar - znear);
393   var d = -2 * zfar * znear / (zfar - znear);
394   
395   return new X.matrix([[x, 0, a, 0], [0, y, b, 0], [0, 0, c, d], [0, 0, -1, 0]]);
396   
397 };

400 /**
401  * Perform a pan operation. This method fires a X.camera.RenderEvent() after the
402  * calculation is done.
403  * 
404  * @param {!goog.math.Vec2} distance The distance of the panning in respect of
405  *          the last camera position.
406  */
407 X.camera.prototype.pan = function(distance) {
408 
409   if (!(distance instanceof goog.math.Vec2)) {
410     
411     throw new X.exception('Fatal: Invalid distance vector for pan operation.');
412     
413   }
414   
415   var distance3d = new goog.math.Vec3(-distance.x, distance.y, 0);
416   
417   var identity = X.matrix.createIdentityMatrix(4);
418   var panMatrix = identity.translate(distance3d);
419   
420   this._view = new X.matrix(panMatrix.multiply(this._view));
421   this._glView = new Float32Array(this._view.flatten());
422   this._glViewInvertedTransposed = new Float32Array(new X.matrix(this._view
423       .getInverse().getTranspose()).flatten());
424   
425   // fire a render event
426   this.dispatchEvent(new X.event.RenderEvent());
427   
428 };

431 /**
432  * Perform a rotate operation. This method fires a X.camera.RenderEvent() after
433  * the calculation is done.
434  * 
435  * @param {!goog.math.Vec2} distance The distance of the rotation in respect of
436  *          the last camera position.
437  */
438 X.camera.prototype.rotate = function(distance) {
439 
440   if (!(distance instanceof goog.math.Vec2)) {
441     
442     throw new X.exception(
443         'Fatal: Invalid distance vector for rotate operation.');
444     
445   }
446   
447   // in radii, the 5 is a constant stating how quick the rotation performs..
448   var angleX = -distance.x / 5 * Math.PI / 180;
449   var angleY = -distance.y / 5 * Math.PI / 180;
450   
451   var identity = X.matrix.createIdentityMatrix(4);
452   // the x-Axis vector is determined by the first row of the view matrix
453   var xAxisVector = new goog.math.Vec3(this._view.getValueAt(0, 0), this._view
454       .getValueAt(0, 1), this._view.getValueAt(0, 2));
455   // the y-Axis vector is determined by the second row of the view matrix
456   var yAxisVector = new goog.math.Vec3(this._view.getValueAt(1, 0), this._view
457       .getValueAt(1, 1), this._view.getValueAt(1, 2));
458   
459   // we rotate around the Y Axis when the mouse moves along the screen in X
460   // direction
461   var rotateX = identity.rotate(angleX, yAxisVector);
462   
463   // we rotate around the X axis when the mouse moves along the screen in Y
464   // direction
465   var rotateY = identity.rotate(angleY, xAxisVector);
466   
467   // perform the actual rotation calculation
468   this._view = new X.matrix(this._view.multiply(rotateY.multiply(rotateX)));
469   this._glView = new Float32Array(this._view.flatten());
470   this._glViewInvertedTransposed = new Float32Array(new X.matrix(this._view
471       .getInverse().getTranspose()).flatten());
472   
473   // fire a render event
474   this.dispatchEvent(new X.event.RenderEvent());
475   
476 };

545 /**
546  * Calculate a view matrix by using the camera position and a focus point.
547  * 
548  * @param {!goog.math.Vec3} cameraPosition The camera position.
549  * @param {!goog.math.Vec3} targetPoint The focus (target) point.
550  * @return {!X.matrix} The view matrix.
551  * @private
552  */
553 X.camera.prototype.lookAt_ = function(cameraPosition, targetPoint) {
554 
555   if (!(cameraPosition instanceof goog.math.Vec3) ||
556       !(targetPoint instanceof goog.math.Vec3)) {
557     
558     throw new X.exception(
559         'Fatal: 3D vectors required for calculating the view.');
560     
561   }
562   
563   var matrix = X.matrix.createIdentityMatrix(4);
564   
565   // Make rotation matrix
566   
567   // Z vector = cameraPosition - targetPoint
568   var zVector = goog.math.Vec3.difference(cameraPosition, targetPoint);
569   
570   // normalize Z
571   zVector = zVector.normalize();
572   
573   // Y vector = up
574   var yVector = this._up.clone();
575   
576   // WARNING: there is a problem if yVector == zVector
577   if (yVector.equals(zVector)) {
578     
579     // we now change the zVector a little bit
580     yVector.z += 0.000001;
581     
582   }
583   
584   // X vector = Y x Z
585   var xVector = goog.math.Vec3.cross(yVector, zVector);
586   
587   // recompute Y vector = Z x X
588   yVector = goog.math.Vec3.cross(zVector, xVector);
589   
590   // normalize X and Y
591   xVector = xVector.normalize();
592   yVector = yVector.normalize();
593   
594   // create view matrix
595   // first row
596   matrix.setValueAt(0, 0, xVector.x);
597   matrix.setValueAt(0, 1, xVector.y);
598   matrix.setValueAt(0, 2, xVector.z);
599   matrix.setValueAt(0, 3, -cameraPosition.x);
600   
601   // second row
602   matrix.setValueAt(1, 0, yVector.x);
603   matrix.setValueAt(1, 1, yVector.y);
604   matrix.setValueAt(1, 2, yVector.z);
605   matrix.setValueAt(1, 3, -cameraPosition.y);
606   
607   // third row
608   matrix.setValueAt(2, 0, zVector.x);
609   matrix.setValueAt(2, 1, zVector.y);
610   matrix.setValueAt(2, 2, zVector.z);
611   matrix.setValueAt(2, 3, -cameraPosition.z);
612   
613   // last row
614   matrix.setValueAt(3, 0, 0);
615   matrix.setValueAt(3, 1, 0);
616   matrix.setValueAt(3, 2, 0);
617   matrix.setValueAt(3, 3, 1);
618   
619   return matrix;
620   
621 };
