 43 /**
 44  * This object loads external files in an asynchronous fashion. In addition, the
 45  * loading process is monitored and summarized to a total progress value.
 46  * 
 47  * @constructor
 48  * @extends X.base
 49  */
 50 X.loader = function() {
 51 
 52   // call the standard constructor of X.base
 53   goog.base(this);
 54   
 55   //
 56   // class attributes
 57   
 58   /**
 59    * @inheritDoc
 60    * @const
 61    */
 62   this._className = 'loader';
 63   
 64   /**
 65    * @private
 66    */
 67   this._jobs_ = null;
 68   
 69   this._progress_ = 0;
 70   
 71 };
 72 // inherit from X.base
 73 goog.inherits(X.loader, X.base);

 76 X.loader.prototype.jobs_ = function() {
 77 
 78   if (!goog.isDefAndNotNull(this._jobs_)) {
 79     
 80     this._jobs_ = new goog.structs.Map();
 81     
 82   }
 83   
 84   return this._jobs_;
 85   
 86 };

 89 X.loader.prototype.completed = function() {
 90 
 91   if (!goog.isDefAndNotNull(this._jobs_)) {
 92     
 93     // there are no jobs (and they never were)
 94     // this is a quick 'jump out'
 95     return true;
 96     
 97   }
 98   
 99   // now we check if all of our jobs are completed
100   return !this._jobs_.containsValue(false);
101   
102 };

104 X.loader.prototype.loadTexture = function(object) {
105 
106   if (!goog.isDefAndNotNull(object.texture())) {
107     
108     throw new X.exception('Internal error during texture loading.');
109     
110   }
111   
112   // setup the image object
113   var image = new Image();
114   var currentTextureFilename = object.texture().file();
115   image.src = currentTextureFilename;
116   
117   // we let the object point to this image
118   object.texture().setImage(image);
119   
120   // handler after the image was completely loaded
121   goog.events.listenOnce(image, goog.events.EventType.LOAD,
122       this.loadTextureCompleted.bind(this, object));
123   
124   // add this loading job to our jobs map
125   this.jobs_().set(object.id(), false);
126   
127   // this is a very fast step so we count it as 30% of the total texture load
128   // time
129   this.addProgress(0.3);
130 };
131 
132 X.loader.prototype.loadTextureCompleted = function(object) {
133 
134   // here we add the rest of the setup step 0.7 and a full progress tick for the
135   // load completion
136   this.addProgress(1.7);
137   
138   setTimeout(function() {
139 
140     // at this point the image for the texture was loaded properly
141     object.texture().setClean();
142     
143     // fire the modified event
144     object.modified();
145     
146     // mark the loading job as completed
147     this.jobs_().set(object.id(), true);
148     
149   }.bind(this), 100);
150   
151 };

153 X.loader.prototype.loadFile = function(object) {
154 
155   if (!goog.isDefAndNotNull(object.file())) {
156     
157     // should not happen :)
158     throw new X.exception('Internal error during file loading.');
159     
160   }
161   
162   // clear all points and normals of the object
163   object.points().clear();
164   object.normals().clear();
165   
166   // get the associated file of the object
167   var filepath = object.file().path();
168   
169   // check if the file is supported
170   var fileExtension = filepath.split('.').pop();
171   fileExtension = fileExtension.toUpperCase();
172   
173   if (!(fileExtension in X.parser.extensions)) {
174     
175     // file format is not supported
176     throw new X.exception('The ' + fileExtension +
177         ' file format is not supported!');
178     
179   }
180   
181   // we use a simple XHR to get the file contents
182   // this works for binary and for ascii files
183   var request = new XMLHttpRequest();
184   
185   // listen to progress events.. here, goog.events.listen did not work
186   // request.addEventListener('progress',
187   // this.loadFileProgress.bind(this, object), false);
188   
189   // listen to abort events
190   goog.events.listen(request, 'abort', this.loadFileFailed.bind(this, request,
191       object));
192   
193   // listen to error events
194   goog.events.listen(request, 'error', this.loadFileFailed.bind(this, request,
195       object));
196   
197   // listen to completed events
198   goog.events.listen(request, 'load', this.loadFileCompleted.bind(this,
199       request, object));
200   
201   // configure the URL
202   request.open('GET', filepath, true);
203   request.overrideMimeType("text/plain; charset=x-user-defined");
204   request.setRequestHeader("Content-Type", "text/plain");
205   
206   // .. and GO!
207   request.send(null);
208   
209   // add this loading job to our jobs map
210   this.jobs_().set(object.id(), false);
211   
212 };

232 X.loader.prototype.addProgress = function(value) {
233 
234   // we have a three stage system during loading
235   //
236   // stage 1: loading in X.loader
237   // stage 2: parsing in X.parser (when loading textures, this does not happen
238   // so the loading adds a 2 in total)
239   // stage 3: setting up in X.renderer
240   //
241   // each stage adds progress from 0..1 with a total of 1 at the end
242   // add a fake job to prevent the user starring at a full progress bar
243   this._progress_ += value / (this.jobs_().getCount()) / 3;
244   
245   if (this._progress_ > 1) {
246     
247     this._progress_ = 1;
248     
249   }
250   
251   var progressEvent = new X.event.ProgressEvent();
252   progressEvent._value = this._progress_;
253   this.dispatchEvent(progressEvent);
254   
255 };

258 X.loader.prototype.progress = function() {
259 
260   return this._progress_;
261   
262 };

265 X.loader.prototype.loadFileFailed = function(request, object) {

271 X.loader.prototype.loadFileCompleted = function(request, object) {
272 
273   // loading completed, add progress
274   this.addProgress(1.0);
275   
276   // we use a timeout here to let the progress bar be able to breath and show
277   // something
278   setTimeout(function() {
279 
280     var filepath = object.file().path();
281     
282     var fileExtension = filepath.split('.').pop();
283     
284     // setup a parser depending on the fileExtension
285     // at this point, we already know that the file format is supported
286     
287     if (fileExtension == 'stl') {
288       
289       var stlParser = new X.parserSTL();
290       
291       goog.events.listenOnce(stlParser, X.event.events.MODIFIED,
292           this.parseFileCompleted.bind(this));
293       
294       stlParser.parse(object, request.response);
295       
296     } else if (fileExtension == 'vtk') {
297       
298       var vtkParser = new X.parserVTK();
299       
300       goog.events.listenOnce(vtkParser, X.event.events.MODIFIED,
301           this.parseFileCompleted.bind(this));
302       
303       vtkParser.parse(object, request.response);
304       
305     } else if (fileExtension == 'trk') {
306       
307       var trkParser = new X.parserTRK();
308       
309       goog.events.listenOnce(trkParser, X.event.events.MODIFIED,
310           this.parseFileCompleted.bind(this));
311       
312       trkParser.parse(object, request.response);
313       
314     }
315     
316 
317   }.bind(this), 100);
318   
319 
320 
321 };

324 X.loader.prototype.parseFileCompleted = function(event) {
325 
326   this.addProgress(1.0);
327   
328   // we use a timeout here to let the progress bar be able to breath and show
329   // something
330   setTimeout(function() {
331 
332     var object = event._object;
333     
334     // the parsing is done here..
335     object.file().setClean();
336     
337     // fire the modified event
338     object.modified();
339     
340     // mark the loading job as completed
341     this.jobs_().set(object.id(), true);
342     
343   }.bind(this), 100);
344   
345 };
