 30 // provides
 31 goog.provide('X.sphere');
 32 
 33 // requires
 34 goog.require('CSG');
 35 goog.require('X.base');
 36 goog.require('X.exception');
 37 goog.require('X.object');
 38 
 39 
 40 
 41 /**
 42  * Create a displayable sphere.
 43  * 
 44  * @constructor
 45  * @param {!Array} center The center position in 3D space as a 1-D Array with
 46  *          length 3.
 47  * @param {!number} radius The radius of the sphere.
 48  * @extends X.object
 49  */
 50 X.sphere = function(center, radius) {
 51 
 52   if (!goog.isDefAndNotNull(center) || !(center instanceof Array) ||
 53       (center.length != 3)) {
 54     
 55     throw new X.exception('Invalid center.');
 56     
 57   }
 58   
 59   if (!goog.isNumber(radius)) {
 60     
 61     throw new X.exception('Invalid radius.');
 62     
 63   }
 64   
 65   //
 66   // call the standard constructor of X.base
 67   goog.base(this);
 68   
 69   //
 70   // class attributes
 71   
 72   /**
 73    * @inheritDoc
 74    * @const
 75    */
 76   this._className = 'sphere';
 77   
 78   this._center = center;
 79   
 80   this._radius = radius;
 81   
 82   this._slices = 32;
 83   
 84   this._stacks = 16;
 85   
 86   this.create_();
 87   
 88 };
 89 // inherit from X.object
 90 goog.inherits(X.sphere, X.object);
 91 
 92 
 93 /**
 94  * Create the sphere.
 95  * 
 96  * @private
 97  */
 98 X.sphere.prototype.create_ = function() {
 99 
100   this.fromCSG(new CSG.sphere({
101     center: this._center,
102     radius: this._radius,
103     slices: this._slices,
104     stacks: this._stacks
105   }));
106   
107 };
108 
109 // export symbols (required for advanced compilation)
110 goog.exportSymbol('X.sphere', X.sphere);
