 30 // provides
 31 goog.provide('X.file');
 32 
 33 // requires
 34 goog.require('X.base');
 35 
 36 
 37 
 38 /**
 39  * Create a container for a files path mapping.
 40  * 
 41  * @param {?string} path The file path for the file.
 42  * @constructor
 43  * @extends X.base
 44  */
 45 X.file = function(path) {
 46 
 47   //
 48   // call the standard constructor of X.base
 49   goog.base(this);
 50   
 51   //
 52   // class attributes
 53   
 54   /**
 55    * @inheritDoc
 56    * @const
 57    */
 58   this['_className'] = 'file';
 59   
 60   /**
 61    * The file path.
 62    * 
 63    * @type {?string}
 64    * @protected
 65    */
 66   this._path = path;
 67   
 68   // mark as dirty since we configure a path here
 69   this._dirty = true;
 70 };
 71 // inherit from X.base
 72 goog.inherits(X.file, X.base);
 73 
 74 
 75 /**
 76  * Get the file path.
 77  * 
 78  * @return {?string} The file path.
 79  */
 80 X.file.prototype.path = function() {
 81 
 82   return this._path;
 83   
 84 };
 85 
 86 
 87 /**
 88  * Set the file path.
 89  * 
 90  * @param {?string} path The file path.
 91  */
 92 X.file.prototype.setPath = function(path) {
 93 
 94   this._path = path;
 95   
 96   // mark as dirty
 97   this._dirty = true;
 98   
 99 };
