 24 /**
 25  * The superclass class for all events in XTK.
 26  * 
 27  * @constructor
 28  * @param {string} type A type identifier for this event.
 29  * @extends goog.events.Event
 30  */
 31 X.event = function(type) {
 32 
 33   //
 34   // call the default event constructor
 35   goog.base(this, type);
 36   
 37   //
 38   // class attributes
 39   
 40   /**
 41    * The className of this class.
 42    * 
 43    * @type {string}
 44    * @protected
 45    */
 46   this._className = 'event';
 47   
 48 };
 49 // inherit from goog.events.Event
 50 goog.inherits(X.event, goog.events.Event);

 53 /**
 54  * Creates a unique event id.
 55  * 
 56  * @param {string} id The id.
 57  * @return {string} A unique id.
 58  */
 59 X.event.uniqueId = function(id) {
 60 
 61   return goog.events.getUniqueId(id);
 62   
 63 };

 65 /**
 66  * The events of this class.
 67  * 
 68  * @enum {string}
 69  */
 70 X.event.events = {
 71   // the pan event, where the event and focus get moved accordingly
 72   PAN: X.event.uniqueId('pan'),
 73   
 74   // the rotate event, where only the event gets moved
 75   ROTATE: X.event.uniqueId('rotate'),
 76   
 77   // the zoom event, where the event Z coordinate changes
 78   ZOOM: X.event.uniqueId('zoom'),
 79   
 80   // the render event
 81   RENDER: X.event.uniqueId('render'),
 82   
 83   // the reset view event
 84   RESETVIEW: X.event.uniqueId('resetview'),
 85   
 86   // the object modified event
 87   MODIFIED: X.event.uniqueId('modified'),
 88   
 89   // the loading progress event
 90   PROGRESS: X.event.uniqueId('progress')
 91 };

150 /**
151  * The zoom event to initiate zoom in or zoom out.
152  * 
153  * @constructor
154  * @extends X.event
155  */
156 X.event.ZoomEvent = function() {
157 
158   // call the default event constructor
159   goog.base(this, X.event.events.ZOOM);
160   
161   /**
162    * The flag for the zooming direction. If TRUE, the zoom operation will move
163    * the objects closer to the event. If FALSE, further away from the event.
164    * 
165    * @type {!boolean}
166    * @protected
167    */
168   this._in = false;
169   
170   /**
171    * The flag for the zooming speed. If TRUE, the zoom operation will happen
172    * fast. If FALSE, there will be a fine zoom operation.
173    * 
174    * @type {!boolean}
175    * @protected
176    */
177   this._fast = false;
178   
179 };
180 // inherit from goog.events.Event
181 goog.inherits(X.event.ZoomEvent, X.event);
