 14 /**
 15  * Create a pair of shaders which consists of a vertex and a fragment shader.
 16  * 
 17  * @constructor
 18  * @extends X.base
 19  */
 20 X.shaders = function() {
 21 
 22   //
 23   // call the standard constructor of X.base
 24   goog.base(this);
 25   
 26   //
 27   // class attributes
 28   
 29   /**
 30    * @inheritDoc
 31    * @const
 32    */
 33   this._className = 'shader';
 34   
 35   /**
 36    * The vertex shader source of this shader pair. By default, a basic shader
 37    * supporting vertex positions and vertex colors is defined.
 38    * 
 39    * @type {!string}
 40    * @protected
 41    */
 42   this._vertexshaderSource = '';
 43   var t = '';
 44   t += 'precision mediump float;\n';
 45   t += '\n';
 46   t += 'attribute vec3 vertexPosition;\n';
 47   t += 'attribute vec3 vertexNormal;\n';
 48   t += 'attribute vec3 vertexColor;\n';
 49   t += 'attribute vec2 vertexTexturePos;\n';
 50   t += '\n';
 51   t += 'uniform mat4 view;\n';
 52   t += 'uniform mat4 perspective;\n';
 53   t += 'uniform mat4 objectTransform;\n';
 54   t += 'uniform bool useObjectColor;\n';
 55   t += 'uniform vec3 objectColor;\n';
 56   t += 'uniform mat4 normal;\n';
 57   t += '\n';
 58   t += 'varying vec4 fVertexPosition;\n';
 59   t += 'varying lowp vec3 fragmentColor;\n';
 60   t += 'varying vec2 fragmentTexturePos;\n';
 61   t += 'varying vec4 fTransformedVertexNormal;\n';
 62   t += '\n';
 63   t += 'void main(void) {\n';
 64   // setup varying -> fragment shader
 65   t += '  fTransformedVertexNormal = normal * objectTransform * vec4(vertexNormal,1.0);\n';
 66   t += '  fVertexPosition = view * objectTransform * vec4(vertexPosition, 1.0);\n';
 67   // t += ' fTransformedVertexNormal = view * vec4(vertexNormal, 1.0);\n';
 68   t += '  fragmentTexturePos = vertexTexturePos;\n';
 69   t += '  if (useObjectColor) {\n';
 70   t += '    fragmentColor = objectColor;\n';
 71   t += '  } else {\n';
 72   t += '    fragmentColor = vertexColor;\n';
 73   t += '  }\n';
 74   // setup vertex Position in the GL context
 75   t += '  gl_Position = perspective * fVertexPosition;\n';
 76   t += '}\n';
 77   this._vertexshaderSource = t;
 78   
 79   /**
 80    * The fragment shader source of this shader pair. By default, a basic shader
 81    * supporting fragment colors is defined.
 82    * 
 83    * @type {!string}
 84    * @protected
 85    */
 86   this._fragmentshaderSource = '';
 87   var t2 = '';
 88   // android only guarantees medium precision
 89   t2 += 'precision mediump float;\n';
 90   t2 += '\n';
 91   t2 += 'uniform bool useTexture;\n';
 92   t2 += 'uniform sampler2D textureSampler;\n';
 93   t2 += 'uniform float objectOpacity;\n';
 94   t2 += '\n';
 95   t2 += 'varying vec4 fVertexPosition;\n';
 96   t2 += 'varying lowp vec3 fragmentColor;\n';
 97   t2 += 'varying vec2 fragmentTexturePos;\n';
 98   t2 += 'varying vec4 fTransformedVertexNormal;\n';
 99   t2 += '\n';
100   t2 += 'void main(void) {\n';
101   t2 += ' if (useTexture) {\n';
102   t2 += '   gl_FragColor = texture2D(textureSampler,';
103   t2 += '   vec2(fragmentTexturePos.s,fragmentTexturePos.t));\n';
104   t2 += ' } else {\n';
105   // configure advanced lighting
106   t2 += '   vec3 nNormal = normalize(fTransformedVertexNormal.xyz);\n';
107   t2 += '   vec3 light = vec3(0.0, 0.0, 1.0);\n';
108   t2 += '   vec3 lightDirection = vec3(-10.0, 4.0, -20.0);\n';
109   t2 += '   lightDirection = normalize(lightDirection);\n';
110   t2 += '   vec3 eyeDirection = normalize(-fVertexPosition.xyz);\n';
111   t2 += ' vec3 reflectionDirection = reflect(-lightDirection, nNormal);\n';
112   // t2 += ' vec3 reflectionDirection = nNormal;\n'; <-- to disable reflection
113   // configure specular (16.0 is material property), diffuse and ambient
114   t2 += '   float specular = pow(max(dot(reflectionDirection, eyeDirection), 0.0), 16.0);\n';
115   t2 += '   float diffuse = 0.8 * max(dot(nNormal, light), 0.0);\n';
116   t2 += '   float ambient = 0.2;\n';
117   // .. and now setup the fragment color using these three values and the
118   // opacity
119   t2 += '   gl_FragColor = vec4(fragmentColor * ambient +\n';
120   t2 += '                       fragmentColor * diffuse +\n';
121   t2 += '                       vec3(0.2, 0.2, 0.2) * specular,\n';
122   t2 += '                       objectOpacity);\n';
123   t2 += ' }\n';
124   t2 += '}\n';
125   this._fragmentshaderSource = t2;
126   
127   /**
128    * The string to access the position inside the vertex shader source.
129    * 
130    * @type {!string}
131    * @protected
132    */
133   this._positionAttribute = 'vertexPosition';
134   
135   /**
136    * The string to access the position inside the vertex shader source.
137    * 
138    * @type {!string}
139    * @protected
140    */
141   this._normalAttribute = 'vertexNormal';
142   
143   /**
144    * The string to access the color inside the vertex shader source.
145    * 
146    * @type {!string}
147    * @protected
148    */
149   this._colorAttribute = 'vertexColor';
150   
151   /**
152    * The string to access the texture position inside the vertex shader source.
153    * 
154    * @type {!string}
155    * @protected
156    */
157   this._texturePosAttribute = 'vertexTexturePos';
158   
159   /**
160    * The string to access the view matrix inside the vertex shader source.
161    * 
162    * @type {!string}
163    * @protected
164    */
165   this._viewUniform = 'view';
166   
167   /**
168    * The string to access the perspective matrix inside the vertex shader
169    * source.
170    * 
171    * @type {!string}
172    * @protected
173    */
174   this._perspectiveUniform = 'perspective';
175   
176 
177   /**
178    * The string to access the transform matrix inside the vertex shader source.
179    * 
180    * @type {!string}
181    * @protected
182    */
183   this._objectTransformUniform = 'objectTransform';
184   
185 
186   this._useObjectColorUniform = 'useObjectColor';
187   
188 
189   /**
190    * The string to access the objectColor uniform inside the vertex shader
191    * source.
192    * 
193    * @type {!string}
194    * @protected
195    */
196   this._objectColorUniform = 'objectColor';
197   
198 
199   /**
200    * The string to access the objectOpacity uniform inside the vertex shader
201    * source.
202    * 
203    * @type {!string}
204    * @protected
205    */
206   this._objectOpacityUniform = 'objectOpacity';
207   
208   // TODO comments
209   
210   this._normalUniform = 'normal';
211   
212   this._useTextureUniform = 'useTexture';
213   
214   this._textureSamplerUniform = 'textureSampler';
215   
216 };
217 // inherit from X.base
218 goog.inherits(X.shaders, X.base);

409 /**
410  * Checks if the configured shaders object is valid.
411  * 
412  * @return {boolean} TRUE or FALSE depending on success.
413  * @throws {X.exception} An exception if the shader is invalid.
414  */
415 X.shaders.prototype.validate = function() {
416 
417   // check if the sources are compatible to the attributes and uniforms
418   
419   var t = 31337;
420   
421   t = this._vertexshaderSource.search(this._positionAttribute);
422   
423   if (t == -1) {
424     
425     throw new X.exception(
426         'Fatal: Could not validate shader! The positionAttribute was bogus.');
427     
428   }
429   
430   t = this._vertexshaderSource.search(this._normalAttribute);
431   
432   if (t == -1) {
433     
434     throw new X.exception(
435         'Fatal: Could not validate shader! The normalAttribute was bogus.');
436     
437   }
438   
439   t = this._vertexshaderSource.search(this._colorAttribute);
440   
441   if (t == -1) {
442     
443     throw new X.exception(
444         'Fatal: Could not validate shader! The colorAttribute was bogus.');
445     
446   }
447   
448   t = this._vertexshaderSource.search(this._opacityAttribute);
449   
450   if (t == -1) {
451     
452     throw new X.exception(
453         'Fatal: Could not validate shader! The opacityAttribute was bogus.');
454     
455   }
456   
457   t = this._vertexshaderSource.search(this._perspectiveUniform);
458   
459   if (t == -1) {
460     
461     throw new X.exception(
462         'Fatal: Could not validate shader! The perspectiveUniform was bogus.');
463     
464   }
465   
466   t = this._vertexshaderSource.search(this._viewUniform);
467   
468   if (t == -1) {
469     
470     throw new X.exception(
471         'Fatal: Could not validate shader! The viewUniform was bogus.');
472     
473   }
474   
475   t = this._vertexshaderSource.search(this._objectTransformUniform);
476   
477   if (t == -1) {
478     
479     throw new X.exception(
480         'Fatal: Could not validate shader! The transformUniform was bogus.');
481     
482   }
483   
484   t = this._vertexshaderSource.search(this._useObjectColorUniform);
485   
486   if (t == -1) {
487     
488     throw new X.exception(
489         'Fatal: Could not validate shader! The useObjectColorUniform was bogus.');
490     
491   }
492   
493   t = this._vertexshaderSource.search(this._objectColorUniform);
494   
495   if (t == -1) {
496     
497     throw new X.exception(
498         'Fatal: Could not validate shader! The objectColorUniform was bogus.');
499     
500   }
501   
502   t = this._fragmentshaderSource.search(this._objectOpacityUniform);
503   
504   if (t == -1) {
505     
506     throw new X.exception(
507         'Fatal: Could not validate shader! The objectOpacityUniform was bogus.');
508     
509   }
510   
511   t = this._vertexshaderSource.search(this._normalUniform);
512   
513   if (t == -1) {
514     
515     throw new X.exception(
516         'Fatal: Could not validate shader! The normalUniform was bogus.');
517     
518   }
519   
520   t = this._vertexshaderSource.search(this._texturePosAttribute);
521   
522   if (t == -1) {
523     
524     throw new X.exception(
525         'Fatal: Could not validate shader! The texturePosAttribute was bogus.');
526     
527   }
528   
529   t = this._fragmentshaderSource.search(this._textureSamplerUniform);
530   
531   if (t == -1) {
532     
533     throw new X.exception(
534         'Fatal: Could not validate shader! The textureSamplerUniform was bogus.');
535     
536   }
537   
538   t = this._fragmentshaderSource.search(this._useTextureUniform);
539   
540   if (t == -1) {
541     
542     throw new X.exception(
543         'Fatal: Could not validate shader! The useTextureUniform was bogus.');
544     
545   }
546   
547   return true;
548
