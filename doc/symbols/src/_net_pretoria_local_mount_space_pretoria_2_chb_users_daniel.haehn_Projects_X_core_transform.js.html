 41 /**
 42  * Create a transform.
 43  * 
 44  * @constructor
 45  * @extends X.base
 46  */
 47 X.transform = function() {
 48 
 49   //
 50   // call the standard constructor of X.base
 51   goog.base(this);
 52   
 53   //
 54   // class attributes
 55   
 56   /**
 57    * @inheritDoc
 58    * @const
 59    */
 60   this._className = 'transform';
 61   
 62   /**
 63    * The transformation matrix.
 64    * 
 65    * @type {!X.matrix}
 66    * @protected
 67    */
 68   this._matrix = X.matrix.createIdentityMatrix(4);
 69   
 70   /**
 71    * The transformation matrix as a 'ready-to-use'-gl version.
 72    * 
 73    * @type {!Object}
 74    * @protected
 75    */
 76   this._glMatrix = new Float32Array(this._matrix.flatten());
 77   
 78 };
 79 // inherit from X.base
 80 goog.inherits(X.transform, X.base);
 81 
 82 
 83 /**
 84  * Rotate around the X-axis.
 85  * 
 86  * @param {number} angle The angle to rotate in degrees.
 87  * @throws {X.exception} An exception, if the given angle is invalid.
 88  */
 89 X.transform.prototype.rotateX = function(angle) {
 90 
 91   if (!goog.isNumber(angle) || angle < -360 || angle > 360) {
 92     
 93     throw new X.exception('Invalid angle!');
 94     
 95   }
 96   
 97   var angleInRadii = angle * Math.PI / 180;
 98   
 99   this._matrix = this._matrix.rotate(angleInRadii, new goog.math.Vec3(0, 1, 0));
100   this._glMatrix = new Float32Array(this._matrix.flatten());
101   
102   this._dirty = true;
103   
104 };

155 /**
156  * Translate on the X-axis.
157  * 
158  * @param {number} distance The distance to move.
159  * @throws {X.exception} An exception, if the given distance is invalid.
160  */
161 X.transform.prototype.translateX = function(distance) {
162 
163   if (!goog.isNumber(distance)) {
164     
165     throw new X.exception('Invalid distance!');
166     
167   }
168   
169   var vector = new goog.math.Vec3(distance, 0, 0);
170   
171   this._matrix = this._matrix.translate(vector);
172   this._glMatrix = new Float32Array(this._matrix.flatten());
173   
174   this._dirty = true;
175   
176 };
