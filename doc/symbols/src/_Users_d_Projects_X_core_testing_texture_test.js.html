 34 /**
 35  * Test for X.texture.file
 36  */
 37 function testXtextureFile() {
 38 
 39   var t = new X.texture();
 40   
 41   // be default, the file should be null
 42   assertEquals(t.file(), null);
 43   
 44   // let's try to set the file by string
 45   t.setFile('/dev/null');
 46   
 47   // .. this should be converted internally to an X.file object
 48   assertTrue(t.file() instanceof X.file);
 49   
 50   // let's try to set the file using X.file directly
 51   var f = new X.file('/dev/null');
 52   
 53   t.setFile(f);
 54   
 55   // .. should be the same
 56   assertEquals(t.file(), f);
 57   
 58 }
