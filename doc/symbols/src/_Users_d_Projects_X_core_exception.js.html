  5 // provides
  6 goog.provide('X.exception');
  7 
  8 // requires
  9 goog.require('X.base');

 13 /**
 14  * The general XTK exception.
 15  * 
 16  * @param {?string} message An error message.
 17  * @constructor
 18  * @extends X.base
 19  */
 20 X.exception = function(message) {
 21 
 22   var validMessage = 'Unknown error!';
 23   
 24   if (goog.isDefAndNotNull(message)) {
 25     
 26     // use the given message
 27     validMessage = message;
 28     
 29   }
 30   
 31   //
 32   // call the standard constructor of X.base
 33   goog.base(this);
 34   
 35   //
 36   // class attributes
 37   
 38   /**
 39    * @inheritDoc
 40    * @const
 41    */
 42   this._className = 'exception';
 43   
 44   /**
 45    * The message of this exception.
 46    * 
 47    * @type {!string}
 48    * @protected
 49    */
 50   this._message = validMessage;
 51   
 52   /**
 53    * The associated error object of this exception.
 54    * 
 55    * @type {Error}
 56    * @private
 57    */
 58   var _err = new Error();
 59   
 60   /**
 61    * The stack trace of this exception.
 62    * 
 63    * @type {!string}
 64    * @private
 65    */
 66   this._stackTrace_ = _err.stack;
 67   
 68 };
 69 goog.inherits(X.exception, X.base);
 70 
 71 // export symbols (required for advanced compilation)
 72 goog.exportSymbol('X.exception', X.exception);
