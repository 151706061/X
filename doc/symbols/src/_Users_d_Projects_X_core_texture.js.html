 30 goog.provide('X.texture');
 31 
 32 // requires
 33 goog.require('X.base');
 34 
 35 
 36 
 37 /**
 38  * Create a texture using a two-dimensional image file.
 39  * 
 40  * @constructor
 41  * @param {string} file The filename of the image.
 42  * @extends X.base
 43  */
 44 X.texture = function(file) {
 45 
 46   if (!goog.isDefAndNotNull(file)) {
 47     
 48     throw new Error('Missing image file for the texture.');
 49     
 50   }
 51   
 52   //
 53   // call the standard constructor of X.base
 54   goog.base(this);
 55   
 56   //
 57   // class attributes
 58   
 59   /**
 60    * @inheritDoc
 61    * @const
 62    */
 63   this['_className'] = 'texture';
 64   
 65   this._file = file;
 66   
 67   this._filter = X.texture.filters.SHARP;
 68   
 69   this._image = null;
 70   
 71   // since we pass a new texture file in the constructor, directly mark this
 72   // object as dirty
 73   this._dirty = true;
 74   
 75 };
 76 // inherit from X.base
 77 goog.inherits(X.texture, X.base);
 78 
 79 
 80 /**
 81  * Different filters for an X.texture.
 82  * 
 83  * @enum {string}
 84  */
 85 X.texture.filters = {
 86   // different filters for texture display
 87   SHARP: 'SHARP',
 88   SMOOTH: 'SMOOTH'
 89 };
 90 
 91 
 92 /**
 93  * Get the image file of this texture.
 94  * 
 95  * @return {!string} The image file of this texture.
 96  */
 97 X.texture.prototype.file = function() {
 98 
 99   return this._file;
100   
101 };
102 
103 
104 X.texture.prototype.image = function() {
105 
106   return this._image;
107   
108 };
109 
110 
111 X.texture.prototype.setImage = function(image) {
112 
113   this._image = image;
114   
115 };
