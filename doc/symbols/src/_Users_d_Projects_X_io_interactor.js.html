 26 /**
 27  * Create an interactor for a given element in the DOM tree.
 28  * 
 29  * @constructor
 30  * @param {Element} element The DOM element to be observed.
 31  * @extends X.base
 32  */
 33 X.interactor = function(element) {
 34 
 35   // check if we have a valid element
 36   if (!goog.isDefAndNotNull(element) || !(element instanceof Element)) {
 37     
 38     throw new X.exception(
 39         'Fatal: Could not add interactor to the given element.');
 40     
 41   }
 42   
 43   //
 44   // call the standard constructor of X.base
 45   goog.base(this);
 46   
 47   //
 48   // class attributes
 49   
 50   /**
 51    * @inheritDoc
 52    * @const
 53    */
 54   this._className = 'interactor';
 55   
 56   /**
 57    * The observed DOM element of this interactor.
 58    * 
 59    * @type {!Element}
 60    * @protected
 61    */
 62   this._element = element;
 63   
 64   /**
 65    * The browser independent mousewheel handler.
 66    * 
 67    * @type {?goog.events.MouseWheelHandler}
 68    * @protected
 69    */
 70   this._mouseWheelHandler = null;
 71   
 72   /**
 73    * Indicates if the keyboard should be observed.
 74    * 
 75    * @type {!boolean}
 76    * @protected
 77    */
 78   this._observeKeyboard = false;
 79   
 80   /**
 81    * Indicates if the mouse is inside the element.
 82    * 
 83    * @type {boolean}
 84    * @protected
 85    */
 86   this._mouseInside = true;
 87   
 88   /**
 89    * Indicates if the left mouse button is pressed.
 90    * 
 91    * @type {boolean}
 92    * @protected
 93    */
 94   this._leftButtonDown = false;
 95   
 96   /**
 97    * Indicates if the middle mouse button is pressed.
 98    * 
 99    * @type {boolean}
100    * @protected
101    */
102   this._middleButtonDown = false;
103   
104   /**
105    * Indicates if the right mouse button is pressed.
106    * 
107    * @type {boolean}
108    * @protected
109    */
110   this._rightButtonDown = false;
111   
112   /**
113    * The previous mouse position.
114    * 
115    * @type {!goog.math.Vec2}
116    * @protected
117    */
118   this._lastMousePosition = new goog.math.Vec2(0, 0);
119   
120 };
121 // inherit from X.base
122 goog.inherits(X.interactor, X.base);

125 /**
126  * Observe mouse wheel interaction on the associated DOM element.
127  */
128 X.interactor.prototype.observeMouseWheel = function() {
129 
130   // we use the goog.events.MouseWheelHandler for a browser-independent
131   // implementation
132   this._mouseWheelHandler = new goog.events.MouseWheelHandler(this._element);
133   
134   goog.events.listen(this._mouseWheelHandler,
135       goog.events.MouseWheelHandler.EventType.MOUSEWHEEL, this.onMouseWheel
136           .bind(this));
137   
138 };

141 /**
142  * Observe mouse clicks on the associated DOM element. This function also blocks
143  * the context menu on right clicks in a browser independent fashion.
144  */
145 X.interactor.prototype.observeMouseClicks = function() {
146 
147   // mouse down
148   goog.events.listen(this._element, goog.events.EventType.MOUSEDOWN,
149       this.onMouseDown.bind(this));
150   
151   // mouse up
152   goog.events.listen(this._element, goog.events.EventType.MOUSEUP,
153       this.onMouseUp.bind(this));
154   
155   // deactivate right-click context menu
156   // found no way to use goog.events for that? tried everything..
157   // according to http://help.dottoro.com/ljhwjsss.php, this method is
158   // compatible with all browsers but opera
159   this._element.oncontextmenu = function() {
160 
161     return false;
162     
163   };
164   
165 };

168 /**
169  * Observe mouse movement on the associated DOM element.
170  */
171 X.interactor.prototype.observeMouseMovement = function() {
172 
173   // mouse movement inside the element
174   goog.events.listen(this._element, goog.events.EventType.MOUSEMOVE,
175       this.onMouseMovementInside.bind(this));
176   
177   // mouse movement outside the element
178   goog.events.listen(this._element, goog.events.EventType.MOUSEOUT,
179       this.onMouseMovementOutside.bind(this));
180   
181 };

184 /**
185  * Observe the keyboard on the associated DOM element.
186  */
187 X.interactor.prototype.observeKeyboard = function() {
188 
189   // set the flag
190   this._observeKeyboard = true;
191   
192   // the google closure way did not work, so let's do it this way..
193   window.onkeydown = this.onKey.bind(this);
194   
195 };

198 /**
199  * Callback for mouse down events on the associated DOM element.
200  * 
201  * @param {Event} event The browser fired event.
202  */
203 X.interactor.prototype.onMouseDown = function(event) {
204 
205   if (event.button == goog.events.BrowserEvent.MouseButton.LEFT) {
206     
207     // left button click
208     this._leftButtonDown = true;
209     
210   } else if (event.button == goog.events.BrowserEvent.MouseButton.MIDDLE) {
211     
212     // middle button click
213     this._middleButtonDown = true;
214     
215   } else if (event.button == goog.events.BrowserEvent.MouseButton.RIGHT) {
216     
217     // right button click
218     this._rightButtonDown = true;
219     
220   }
221   
222   // prevent further handling by the browser
223   event.preventDefault();
224   
225 };

228 /**
229  * Callback for mouse up events on the associated DOM element.
230  * 
231  * @param {Event} event The browser fired event.
232  */
233 X.interactor.prototype.onMouseUp = function(event) {
234 
235   if (event.button == goog.events.BrowserEvent.MouseButton.LEFT) {
236     
237     // left button click
238     this._leftButtonDown = false;
239     
240   } else if (event.button == goog.events.BrowserEvent.MouseButton.MIDDLE) {
241     
242     // middle button click
243     this._middleButtonDown = false;
244     
245   } else if (event.button == goog.events.BrowserEvent.MouseButton.RIGHT) {
246     
247     // right button click
248     this._rightButtonDown = false;
249     
250   }
251   
252   // prevent further handling by the browser
253   event.preventDefault();
254   
255 };

258 /**
259  * Callback for mouse movement events outside the associated DOM element. This
260  * resets all internal interactor flags.
261  * 
262  * @param {Event} event The browser fired event.
263  */
264 X.interactor.prototype.onMouseMovementOutside = function(event) {
265 
266   // reset the click flags
267   this._mouseInside = false;
268   if (this._observeKeyboard) {
269     
270     // if we observe the keyboard, remove the observer here
271     // this is necessary if there are more than one renderer in the document
272     window.onkeydown = null;
273     
274   }
275   
276   this._leftButtonDown = false;
277   this._middleButtonDown = false;
278   this._rightButtonDown = false;
279   this._lastMousePosition.x = 0;
280   this._lastMousePosition.y = 0;
281   
282   // prevent further handling by the browser
283   event.preventDefault();
284   
285 };

288 /**
289  * Callback for mouse movement events inside the associated DOM element. This
290  * distinguishes by pressed mouse buttons, key accelerators etc. and fires
291  * proper X.event events.
292  * 
293  * @param {Event} event The browser fired event.
294  */
295 X.interactor.prototype.onMouseMovementInside = function(event) {
296 
297   // TODO this needs to be more generalized
298   this.dispatchEvent('mouseup');
299   
300   this._mouseInside = true;
301   
302   if (this._observeKeyboard && window.onkeydown == null) {
303     
304     this.observeKeyboard();
305     
306   }
307   
308   // prevent any other actions by the browser (f.e. scrolling, selection..)
309   event.preventDefault();
310   
311   // is shift down?
312   var shiftDown = event.shiftKey;
313   
314   // grab the current mouse position
315   var currentMousePosition = new goog.math.Vec2(event.clientX, event.clientY);
316   
317   // get the distance in terms of the last mouse move event
318   var distance = this._lastMousePosition.subtract(currentMousePosition);
319   
320   // save the current mouse position as the last one
321   this._lastMousePosition = currentMousePosition.clone();
322   
323   // threshold the distance to avoid 'irregular' movement
324   if (Math.abs(distance.x) < 2) {
325     
326     distance.x = 0;
327     
328   }
329   if (Math.abs(distance.y) < 2) {
330     
331     distance.y = 0;
332     
333   }
334   
335   // jump out if the distance is 0 to avoid unnecessary events
336   if (distance.magnitude() == 0) {
337     
338     return;
339     
340   }
341   
342 
343   //
344   // check which mouse buttons or keys are pressed
345   //
346   if (this._leftButtonDown && !shiftDown) {
347     //
348     // LEFT MOUSE BUTTON DOWN AND NOT SHIFT DOWN
349     //
350     
351     // create a new rotate event
352     var e = new X.event.RotateEvent();
353     
354     // attach the distance vector
355     e._distance = distance;
356     
357     // attach the angle in degrees
358     e._angle = 0;
359     
360     // .. fire the event
361     this.dispatchEvent(e);
362     
363 
364   } else if (this._middleButtonDown || (this._leftButtonDown && shiftDown)) {
365     //
366     // MIDDLE MOUSE BUTTON DOWN or LEFT MOUSE BUTTON AND SHIFT DOWN
367     //
368     
369     // create a new pan event
370     var e = new X.event.PanEvent();
371     
372     // panning in general moves pretty fast, so we threshold the distance
373     // additionally
374     if (distance.x > 5) {
375       
376       distance.x = 5;
377       
378     } else if (distance.x < -5) {
379       
380       distance.x = -5;
381       
382     }
383     if (distance.y > 5) {
384       
385       distance.y = 5;
386       
387     } else if (distance.y < -5) {
388       
389       distance.y = -5;
390       
391     }
392     
393     // attach the distance vector
394     e._distance = distance;
395     
396     // .. fire the event
397     this.dispatchEvent(e);
398     
399 
400   } else if (this._rightButtonDown) {
401     //
402     // RIGHT MOUSE BUTTON DOWN
403     //
404     
405     // create a new zoom event
406     var e = new X.event.ZoomEvent();
407     
408     // set the zoom direction
409     // true if zooming in, false if zooming out
410     e._in = (distance.y > 0);
411     
412     // with the right click, the zoom will happen rather
413     // fine than fast
414     e._fast = false;
415     
416     // .. fire the event
417     this.dispatchEvent(e);
418     
419 
420   }
421   
422 };

425 /**
426  * Callback for mouse wheel events on the associated DOM element. This fires
427  * proper X.event events.
428  * 
429  * @param {Event} event The browser fired event.
430  */
431 X.interactor.prototype.onMouseWheel = function(event) {
432 
433   // prevent any other action (like scrolling..)
434   event.preventDefault();
435   
436   // create a new zoom event
437   var e = new X.event.ZoomEvent();
438   
439   // set the zoom direction
440   // true if zooming in, false if zooming out
441   // delta is here given by the event
442   e._in = (event.deltaY < 0);
443   
444   // with the mouseWheel, the zoom will happen rather
445   // fast than fine
446   e._fast = true;
447   
448   // .. fire the event
449   this.dispatchEvent(e);
450   
451 };

454 /**
455  * Callback for keyboard events on the associated DOM element. This fires proper
456  * X.event events.
457  * 
458  * @param {Event} event The browser fired event.
459  */
460 X.interactor.prototype.onKey = function(event) {
461 
462   // only listen to key events if the mouse is inside our element
463   // this f.e. enables key event listening for multiple renderers
464   if (!this._mouseInside) {
465     
466     return;
467     
468   }
469   
470   // observe the control keys (shift, alt, ..)
471   var alt = event.altKey;
472   var ctrl = event.ctrlKey;
473   var meta = event.metaKey; // this is f.e. the windows or apple key
474   var shift = event.shiftKey;
475   
476   // get the keyCode
477   var keyCode = event.keyCode;
478   
479   if (keyCode == 82 && !alt && !ctrl && !meta && !shift) {
480     
481     // 'r' but without any other control keys since we do not want to limit the
482     // user to press for example CTRL+R to reload the page
483     
484     // prevent any other actions..
485     event.preventDefault();
486     
487     // fire the ResetViewEvent
488     var e = new X.event.ResetViewEvent();
489     this.dispatchEvent(e);
490     
491   } else if (keyCode >= 37 && keyCode <= 40) {
492     
493     // keyCode <= 37 and >= 40 means the arrow keys
494     
495     // prevent any other actions..
496     event.preventDefault();
497     
498     var e = null;
499     
500     if (shift) {
501       
502       // create a new pan event
503       e = new X.event.PanEvent();
504       
505     } else if (alt) {
506       
507       // create a new zoom event
508       e = new X.event.ZoomEvent();
509       
510     } else {
511       // create a new rotate event
512       e = new X.event.RotateEvent();
513       
514     }
515     
516     if (!e) {
517       
518       // should not happen but you never know with key interaction
519       return;
520       
521     }
522     
523     // create a distance vector
524     var distance = new goog.math.Vec2(0, 0);
525     
526     if (keyCode == 37) {
527       // '<-' LEFT
528       distance.x = 5;
529       if (alt) {
530         // for zoom, we configure the zooming behavior
531         e._in = true;
532         e._fast = false;
533       }
534       
535     } else if (keyCode == 39) {
536       // '->' RIGHT
537       distance.x = -5;
538       if (alt) {
539         // for zoom, we configure the zooming behavior
540         e._in = false;
541         e._fast = false;
542       }
543       
544     } else if (keyCode == 38) {
545       // '^-' TOP
546       distance.y = 5;
547       if (alt) {
548         // for zoom, we configure the zooming behavior
549         e._in = true;
550         e._fast = true;
551       }
552       
553     } else if (keyCode == 40) {
554       // '-v' BOTTOM
555       distance.y = -5;
556       if (alt) {
557         // for zoom, we configure the zooming behavior
558         e._in = false;
559         e._fast = true;
560       }
561       
562     }
563     
564     // attach the distance vector
565     e._distance = distance;
566     
567     // .. fire the event
568     this.dispatchEvent(e);
569     
570   }
571   
572 };
