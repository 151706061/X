 60 //
 61 // 1. Safari does not support the .bind(this) functionality which is crucial for
 62 // XTK's event mechanism. This hack fixes this.
 63 //
 64 if (!Function.prototype.bind) {
 65   Function.prototype.bind = function(oThis) {
 66 
 67     if (typeof this !== "function") {
 68       // closest thing possible to the ECMAScript 5 internal IsCallable function
 69       throw new TypeError(
 70           "Function.prototype.bind - what is trying to be bound is not callable");
 71     }
 72     
 73     var fSlice = Array.prototype.slice, aArgs = fSlice.call(arguments, 1), fToBind = this;
 74     
 75     /**
 76      * @constructor
 77      */
 78     var fNOP = function() {
 79 
 80     };
 81     
 82     var fBound = function() {
 83 
 84       return fToBind.apply(this instanceof fNOP ? this : oThis || window, aArgs
 85           .concat(fSlice.call(arguments)));
 86     };
 87     
 88     fNOP.prototype = this.prototype;
 89     fBound.prototype = new fNOP();
 90     
 91     return fBound;
 92   };
 93 }
 94 
 95 goog.exportSymbol('Function.prototype.bind', Function.prototype.bind);
