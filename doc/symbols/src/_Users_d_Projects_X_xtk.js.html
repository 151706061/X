 34 // 1. Safari does not support the .bind(this) functionality which is crucial for
 35 // XTK's event mechanism. This hack fixes this.

 37 if (!Function.prototype.bind) {
 38   Function.prototype.bind = function(oThis) {
 39 
 40     if (typeof this !== "function") {
 41       // closest thing possible to the ECMAScript 5 internal IsCallable function
 42       throw new TypeError(
 43           "Function.prototype.bind - what is trying to be bound is not callable");
 44     }
 45     
 46     var fSlice = Array.prototype.slice, aArgs = fSlice.call(arguments, 1), fToBind = this, fNOP = function() {
 47 
 48     }, fBound = function() {
 49 
 50       return fToBind.apply(this instanceof fNOP ? this : oThis || window, aArgs
 51           .concat(fSlice.call(arguments)));
 52     };
 53     
 54     fNOP.prototype = this.prototype;
 55     fBound.prototype = new fNOP();
 56     
 57     return fBound;
 58   };
