 50 /**
 51  * Create a displayable object. Objects may have points, colors, a texture, or
 52  * may be loaded from a file in addition to opacity and visibility settings. If
 53  * another X.object is passed to this constructor, the properties from this
 54  * X.object are used to configure the new one.
 55  * 
 56  * @constructor
 57  * @param {X.object=} object Another X.object to use as a template.
 58  * @extends X.base
 59  */
 60 X.object = function(object) {
 61 
 62   //
 63   // call the standard constructor of X.base
 64   goog.base(this);
 65   
 66   //
 67   // class attributes
 68   
 69   /**
 70    * @inheritDoc
 71    * @const
 72    */
 73   this['_className'] = 'object';
 74   
 75   // the global id counter
 76   var counter = window["X.Counter"];
 77   // ..get a new unique id
 78   counter.increment();
 79   
 80   /**
 81    * The uniqueId of this object. Each object in XTK has a uniqueId.
 82    * 
 83    * @type {number}
 84    * @protected
 85    */
 86   this['_id'] = counter.value();
 87   
 88   /**
 89    * The rendering type of this object, default is {X.object.types.TRIANGLES}.
 90    * 
 91    * @type {X.object.types}
 92    * @protected
 93    */
 94   this['_type'] = X.object.types.TRIANGLES;
 95   
 96   /**
 97    * The transform of this object.
 98    * 
 99    * @type {!X.transform}
100    * @protected
101    */
102   this._transform = new X.transform();
103   
104   /**
105    * The object color. By default, this is white.
106    * 
107    * @type {!Array}
108    * @public
109    */
110   this['_color'] = [1, 1, 1];
111   
112   /**
113    * The points of this object.
114    * 
115    * @type {!X.triplets}
116    * @protected
117    */
118   this._points = new X.triplets();
119   
120   /**
121    * The normals of this object.
122    * 
123    * @type {!X.triplets}
124    * @protected
125    */
126   this._normals = new X.triplets();
127   
128   /**
129    * The point colors of this object.
130    * 
131    * @type {!X.triplets}
132    * @protected
133    */
134   this._colors = new X.triplets();
135   
136   /**
137    * The texture of this object.
138    * 
139    * @type {?X.texture}
140    * @protected
141    */
142   this._texture = null;
143   
144   /**
145    * The mapping between object and texture coordinates.
146    * 
147    * @type {?Array}
148    * @protected
149    */
150   this._textureCoordinateMap = null;
151   
152   /**
153    * The file of this object.
154    * 
155    * @type {?X.file}
156    * @protected
157    */
158   this._file = null;
159   
160   /**
161    * The opacity of this object.
162    * 
163    * @type {number}
164    * @public
165    */
166   this['_opacity'] = 1.0;
167   
168   /**
169    * The children of this object.
170    * 
171    * @type {?Array}
172    * @protected
173    */
174   this._children = null;
175   
176   /**
177    * The visibility of this object.
178    * 
179    * @type {boolean}
180    * @public
181    */
182   this['_visible'] = true;
183   
184   /**
185    * The point size, only used in X.object.types.POINTS mode.
186    * 
187    * @type {number}
188    * @protected
189    */
190   this['_pointSize'] = 1;
191   
192   /**
193    * The line width, only used in X.object.types.LINES mode.
194    * 
195    * @type {number}
196    * @protected
197    */
198   this['_lineWidth'] = 1;
199   
200   /**
201    * The caption of this object.
202    * 
203    * @type {?string}
204    * @protected
205    */
206   this['_caption'] = null;
207   
208   /**
209    * The flag for the magic mode.
210    * 
211    * @type {!boolean}
212    * @protected
213    */
214   this['_magicMode'] = false;
215   
216   if (goog.isDefAndNotNull(object)) {
217     
218     // copy the properties of the given object over
219     this.copy_(object);
220     
221   }
222   
223   /**
224    * The color table of this object.
225    * 
226    * @type {?X.colorTable}
227    * @protected
228    */
229   this._colorTable = null;
230   
231   /**
232    * The scalars of this object.
233    * 
234    * @type {?X.scalars}
235    * @protected
236    */
237   this._scalars = null;
238   
239   /**
240    * An array reflecting the point or vertex indices.
241    * 
242    * @type {!Array}
243    * @protected
244    */
245   this._pointIndices = [];
246   
247   this._dirty = true;
248   
249 };

269 /**
270  * Copies the properties from a given object to this object. The texture,
271  * textureCoordinateMap and the children are not copied but linked.
272  * 
273  * @param {!X.object} object The given object.
274  * @private
275  */
276 X.object.prototype.copy_ = function(object) {
277 
278   this['_type'] = object.type();
279   
280   this._transform.setMatrix(new X.matrix(object.transform().matrix()));
281   
282   this['_color'] = new Array(object.color());
283   
284   this._points = new X.triplets(object.points());
285   
286   this._normals = new X.triplets(object.normals());
287   
288   this._colors = new X.triplets(object.colors());
289   
290   // do we need to copy this? maybe not
291   this._texture = object.texture();
292   this._textureCoordinateMap = object.textureCoordinateMap();
293   
294   if (object.file()) {
295     // only if a file is configured
296     this._file = new X.file(new String(object.file().path()).toString());
297   }
298   
299   this['_opacity'] = object.opacity();
300   
301   // note: children are not copied
302   this._children = object.children();
303   
304   this['_visible'] = object.visible();
305   
306   this['_pointSize'] = object.pointSize();
307   
308   this['_lineWidth'] = object.lineWidth();
309   
310   if (object.caption()) {
311     // only if a caption is configured
312     this['_caption'] = new String(object.caption()).toString();
313   }
314   
315   this['_magicMode'] = object.magicMode();
316   
317   this._dirty = true;
318   
319 };

334 X.object.prototype.toCSG = function() {
335 
336   var numberOfPoints = this._points.count();
337   
338   var polygons = [];
339   
340   var p = 0;
341   for (p = 0; p < numberOfPoints; p = p + 3) {
342     
343     var point1 = this._points.get(p);
344     var point2 = this._points.get(p + 1);
345     var point3 = this._points.get(p + 2);
346     
347     var normal1 = this._normals.get(p);
348     var normal2 = this._normals.get(p + 1);
349     var normal3 = this._normals.get(p + 2);
350     
351     // get the object color
352     var color = this['_color'];
353     
354     // if point colors are defined on this X.object, use'em
355     if ((this._colors.length() > 0)) {
356       
357       // we only grab the color of the first point since CSG only supports
358       // colors per triangle not per point
359       color = this._colors.get(p);
360       
361     }
362     
363     //
364     // create a new CSG.Polygon
365     //
366     
367     var vertices = [];
368     vertices.push(new csgVertex(point1, normal1));
369     vertices.push(new csgVertex(point2, normal2));
370     vertices.push(new csgVertex(point3, normal3));
371     
372     polygons.push(new csgPolygon(vertices, color));
373     
374   }
375   
376   //
377   // create and return a new CSG object
378   return CSG.fromPolygons(polygons);
379   
380 };

383 X.object.prototype.fromCSG = function(csg) {
384 
385   if (!goog.isDefAndNotNull(csg) || !(csg instanceof CSG)) {
386     
387     throw new Error('Invalid CSG object.');
388     
389   }
390   
391   // remove all previous points
392   this._points.clear();
393   this._normals.clear();
394   this._colors.clear();
395   
396   var indexer = new X.indexer();
397   
398   // .. a temp. array to store the triangles using vertex indices
399   var triangles = new Array();
400   // grab points, normals and colors
401   var csg2poly = csg.toPolygons();
402   goog.array.map(csg2poly, function(p) {
403 
404     var indices = new Array();
405     var ver = p.vertices();
406     var shared = p.shared();
407     indices = goog.array.map(ver, function(vertex) {
408 
409       vertex.color = shared;
410       return indexer.add(vertex);
411     });
412     
413     var i = 2;
414     for (i = 2; i < indices.length; i++) {
415       triangles.push([indices[0], indices[i - 1], indices[i]]);
416     }
417     
418   }.bind(this));
419   
420   this.__vertices = goog.array.map(indexer.unique(), function(v) {
421 
422     return [v.pos().x(), v.pos().y(), v.pos().z()];
423   });
424   
425   this.__normals = goog.array.map(indexer.unique(), function(v) {
426 
427     return [v.normal().x(), v.normal().y(), v.normal().z()];
428   });
429   this.__colors = goog.array.map(indexer.unique(), function(v) {
430 
431     if (!v.color) {
432       
433       return null;
434       
435     }
436     return [v.color[0], v.color[1], v.color[2]];
437   });
438   
439   //
440   // setup the points, normals and colors for this X.object
441   // by converting the triangles to the X.object API
442   goog.array.map(triangles, function(i) {
443 
444     // grab the three vertices of this triangle
445     var i0 = i[0];
446     var i1 = i[1];
447     var i2 = i[2];
448     
449     var vertices = this.__vertices;
450     var normals = this.__normals;
451     var colors = this.__colors;
452     
453     // add the points
454     this._points.add(vertices[i0][0], vertices[i0][1], vertices[i0][2]);
455     this._points.add(vertices[i1][0], vertices[i1][1], vertices[i1][2]);
456     this._points.add(vertices[i2][0], vertices[i2][1], vertices[i2][2]);
457     
458     // add the normals
459     this._normals.add(normals[i0][0], normals[i0][1], normals[i0][2]);
460     this._normals.add(normals[i1][0], normals[i1][1], normals[i1][2]);
461     this._normals.add(normals[i2][0], normals[i2][1], normals[i2][2]);
462     
463     // if colors are set for this triangle, add'em
464     if (colors[i0]) {
465       this._colors.add(colors[i0][0], colors[i0][1], colors[i0][2]);
466     }
467     if (colors[i1]) {
468       this._colors.add(colors[i1][0], colors[i1][1], colors[i1][2]);
469     }
470     if (colors[i2]) {
471       this._colors.add(colors[i2][0], colors[i2][1], colors[i2][2]);
472     }
473     
474   }.bind(this));
475   
476   // we only support CSG in TRIANGLES rendering mode
477   this.setType(X.object.types.TRIANGLES);
478   
479 };

579 /**
580  * Set the object texture. If null is passed, the object will have no texture.
581  * 
582  * @param {?X.texture|string} texture The new texture or an image file path.
583  * @throws {Error} An exception if the given texture is invalid.
584  */
585 X.object.prototype.setTexture = function(texture) {
586 
587   if (!goog.isDefAndNotNull(texture)) {
588     
589     // null textures are allowed
590     this._texture = null;
591     return;
592     
593   }
594   
595   if (goog.isString(texture)) {
596     
597     // a string has to be converted to a new X.texture
598     var textureFile = texture;
599     texture = new X.texture();
600     texture.setFile(textureFile);
601     
602   }
603   
604   if (!(texture instanceof X.texture)) {
605     
606     throw new Error('Invalid texture.');
607     
608   }
609   
610   this._texture = texture;
611   
612 };

675 /**
676  * Set the object color. This overrides any point colors.
677  * 
678  * @param {!number} r The Red value in the range of 0..1
679  * @param {!number} g The Green value in the range of 0..1
680  * @param {!number} b The Blue value in the range of 0..1
681  * @throws {Error} An exception if the given color values are invalid.
682  */
683 X.object.prototype.setColor = function(r, g, b) {
684 
685   // we accept only numbers as arguments
686   if ((!goog.isNumber(r) && r < 0.0 && r > 1.0) ||
687       (!goog.isNumber(g) && g < 0.0 && g > 1.0) ||
688       (!goog.isNumber(b) && b < 0.0 && b > 1.0)) {
689     
690     throw new Error('Invalid color.');
691     
692   }
693   
694   if (this.hasChildren()) {
695     
696     // loop through the children and propagate the new color
697     var children = this.children();
698     var numberOfChildren = children.length;
699     var c = 0;
700     
701     for (c = 0; c < numberOfChildren; c++) {
702       
703       children[c].setColor(r, g, b);
704       
705     }
706     
707   }
708   
709   this['_color'][0] = r;
710   this['_color'][1] = g;
711   this['_color'][2] = b;
712   
713   this._dirty = true;
714   
715 };

718 X.object.prototype.union = function(object) {
719 
720   if (!goog.isDefAndNotNull(object) ||
721       (!(object instanceof CSG) && !(object instanceof X.object))) {
722     
723     throw new Error('Invalid object.');
724     
725   }
726   
727   var csg = object;
728   
729   // if we operate on another X.object, we gotta convert
730   if (object instanceof X.object) {
731     
732     csg = csg.toCSG();
733     
734   }
735   
736   var result = new X.object();
737   result.fromCSG(this.toCSG().union(csg));
738   
739   return result;
740   
741 };

903  * Set the opacity of this object.
904  * 
905  * @param {number} opacity The opacity value in the range 0..1.
906  */
907 X.object.prototype.setOpacity = function(opacity) {
908 
909   // check if the given opacity is in the range 0..1
910   if (!goog.isNumber(opacity) || opacity > 1.0 || opacity < 0.0) {
911     
912     throw new Error('Invalid opacity.');
913     
914   }
915   
916   if (this.hasChildren()) {
917     
918     // loop through the children and propagate the new opacity
919     var children = this.children();
920     var numberOfChildren = children.length;
921     var c = 0;
922     
923     for (c = 0; c < numberOfChildren; c++) {
924       
925       children[c].setOpacity(opacity);
926       
927     }
928     
929   }
930   
931   this['_opacity'] = opacity;
932   
933   this._dirty = true;
934   
935 };

1171 /**
1172  * Compare two X.objects by their opacity values and their distance to the
1173  * viewer's eye. Fully opaque objects should be always ordered before
1174  * transparent ones, and the transparent ones should be ordered back-to-front in
1175  * terms of the distance to the viewer's eye.
1176  * 
1177  * @param {X.object} object1 Object1 to compare against Object2.
1178  * @param {X.object} object2 Object2 to compare against Object1.
1179  * @return {!number} 1, if Object1 should be ordered after Object2 Object2. -1,
1180  *         if Object1 should be ordered before Object2
1181  */
1182 X.object.OPACITY_COMPARATOR = function(object1, object2) {
1183 
1184   // check if we have two valid objects to compare
1185   if (!goog.isDefAndNotNull(object1) || !goog.isDefAndNotNull(object2) ||
1186       !(object1 instanceof X.object) || !(object2 instanceof X.object)) {
1187     
1188     throw new Error('Fatal: Two valid X.objects are required for comparison.');
1189     
1190   }
1191   
1192   // full opaque objects should always be rendered first
1193   if (object1.opacity() == 1) {
1194     
1195     // always put object1 before object2
1196     return -1;
1197     
1198   }
1199   if (object2.opacity() == 1) {
1200     
1201     // always put object2 before object1
1202     return 1;
1203     
1204   }
1205   
1206   if (goog.isDefAndNotNull(object1.distance) &&
1207       goog.isDefAndNotNull(object2.distance)) {
1208     
1209     // order back-to-front from the viewer's eye
1210     
1211     if (object1.distance > object2.distance) {
1212       
1213       // object2 is closer so object1 should be ordered (drawn) before object2
1214       return -1;
1215       
1216     } else if (object1.distance <= object2.distance) {
1217       
1218       // object 1 is closer so object1 should be ordered (drawn) after object2
1219       return 1;
1220       
1221     }
1222     
1223 
1224   }
1225   
1226   return 1;
1227   
1228 };
