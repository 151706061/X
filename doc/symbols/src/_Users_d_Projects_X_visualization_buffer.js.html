  5 // provides
  6 goog.provide('X.buffer');

  8 // requires
  9 goog.require('X.base');
 10 goog.require('X.exception');

 12 
 13 
 14 /**
 15  * Create a GL buffer container. Besides the actual GL buffer, the container
 16  * stores the number of items and the size of each item.
 17  * 
 18  * @constructor
 19  * @param {Object} glBuffer The GL buffer.
 20  * @param {number} itemCount The number of items.
 21  * @param {number} itemSize The size of each item.
 22  * @extends X.base
 23  */
 24 X.buffer = function(glBuffer, itemCount, itemSize) {
 25 
 26   if (!goog.isDefAndNotNull(glBuffer)) {
 27     
 28     throw new X.exception('Fatal: Invalid GL Buffer!');
 29     
 30   }
 31   
 32   if (!goog.isDefAndNotNull(itemCount)) {
 33     
 34     throw new X.exception('Fatal: Invalid number of items!');
 35     
 36   }
 37   
 38   if (!goog.isDefAndNotNull(itemSize)) {
 39     
 40     throw new X.exception('Fatal: Invalid item size!');
 41     
 42   }
 43   
 44   // call the standard constructor of X.base
 45   goog.base(this);
 46   
 47   //
 48   // class attributes
 49   
 50   /**
 51    * @inheritDoc
 52    * @const
 53    */
 54   this._className = 'buffer';
 55   
 56   /**
 57    * The GL buffer.
 58    * 
 59    * @type {Object}
 60    * @const
 61    */
 62   this._glBuffer = glBuffer;
 63   
 64   /**
 65    * The number of items.
 66    * 
 67    * @type {number}
 68    * @const
 69    */
 70   this._itemCount = itemCount;
 71   
 72   /**
 73    * The size of each item.
 74    * 
 75    * @type {number}
 76    * @const
 77    */
 78   this._itemSize = itemSize;
 79   
 80 };
 81 // inherit from X.base
 82 goog.inherits(X.buffer, X.base);
 83 
 84 
 85 /**
 86  * Get the actual GL buffer.
 87  * 
 88  * @return {Object} The GL buffer.
 89  */
 90 X.buffer.prototype.glBuffer = function() {
 91 
 92   return this._glBuffer;
 93   
 94 };
 95 
 96 
 97 /**
 98  * Get the number of the items in this buffer.
 99  * 
100  * @return {number} The number of the items.
101  */
102 X.buffer.prototype.itemCount = function() {
103 
104   return this._itemCount;
105   
106 };
107 
108 
109 /**
110  * Get the size of the items in this buffer.
111  * 
112  * @return {number} The size of the items.
113  */
114 X.buffer.prototype.itemSize = function() {
115 
116   return this._itemSize;
117   
118 };
