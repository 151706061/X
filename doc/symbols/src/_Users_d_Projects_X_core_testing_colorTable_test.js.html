 35 /**
 36  * Test for X.colorTable.file
 37  */
 38 function testXcolorTableFile() {
 39 
 40   var c = new X.colorTable();
 41   
 42   // be default, the file should be null
 43   assertEquals(c.file(), null);
 44   
 45   // let's try to set the file by string
 46   c.setFile('/dev/null');
 47   
 48   // .. this should be converted internally to an X.file object
 49   assertTrue(c.file() instanceof X.file);
 50   
 51   // let's try to set the file using X.file directly
 52   var f = new X.file('/dev/null');
 53   
 54   c.setFile(f);
 55   
 56   // .. should be the same
 57   assertEquals(c.file(), f);
 58   
 59 }

 61 /**
 62  * Test for X.colorTable.add
 63  */
 64 function testXcolorTableAdd() {
 65 
 66   var c = new X.colorTable();
 67   
 68   // a fresh colorTable should not be dirty
 69   assertFalse(c.dirty());
 70   
 71   // define an example color table entry
 72   var colorValue = 0;
 73   var colorLabel = "Label value 0";
 74   var colorR = 255;
 75   var colorG = 254;
 76   var colorB = 253;
 77   var colorA = 128;
 78   
 79   // add the example color
 80   c.add(colorValue, colorLabel, colorR, colorG, colorB, colorA);
 81   
 82   // .. now grab the entry using the colorValue
 83   var returnedColor = c._map.get(colorValue);
 84   
 85   assertEquals(returnedColor[0], colorLabel);
 86   assertEquals(returnedColor[1], colorR);
 87   assertEquals(returnedColor[2], colorG);
 88   assertEquals(returnedColor[3], colorB);
 89   assertEquals(returnedColor[4], colorA);
 90   
 91 }
