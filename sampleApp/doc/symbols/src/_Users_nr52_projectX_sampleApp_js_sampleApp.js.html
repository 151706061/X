1 goog.provide('sampleApp');
  2 goog.provide('sampleApp.run');
  3 
  4 goog.require('X.console');
  5 goog.require('X.renderer2D');
  6 goog.require('X.renderer3D');
  7 
  8 // provides extern
  9 sampleApp.run = function() {
 10 
 11   var c = new X.console();
 12   
 13   c.out('Starting sampleApp..');
 14   
 15   try {
 16     
 17     // create a red 2D renderer in the div with id '2d' (see ../index.html)
 18     var r2d = new X.renderer2D(100, 100);
 19     r2d.setContainerById('2d');
 20     r2d.setBackgroundColor('#ff0000');
 21     r2d.init();
 22     c.out(r2d.print());
 23 
 24     // create a green 3D renderer in the div with id '3d' (see ../index.html)
 25     var r3d = new X.renderer3D(200, 200);
 26     r3d.setContainerById('3d');
 27     r3d.setBackgroundColor('green');
 28     r3d.init();    
 29     c.out(r3d.print());
 30     
 31     // create a default colored (black) 2D renderer without container specification which should
 32     // create it directly in <body></body>
 33     var r2dWithoutHome = new X.renderer2D(10,10);
 34     r2dWithoutHome.init();
 35     c.out(r2dWithoutHome.print());
 36     
 37   } catch (e) {
 38     
 39     // catch any X.exception and print the error to the error log
 40     
 41     if (e instanceof X.exception) {
 42       c.err(e.print());
 43     }
 44     
 45   }  
 46   
 47 };
 48 
 49 // export symbols (requiered for advanced compilation)
 50 goog.exportSymbol('sampleApp', sampleApp);
 51 goog.exportSymbol('sampleApp.run', sampleApp.run);
